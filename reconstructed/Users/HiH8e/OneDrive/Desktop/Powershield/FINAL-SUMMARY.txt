🔧 TROUBLESHOOTING
═══════════════════════════════════════════════════════════════════════════════

Problem: "Ollama connection failed"
Solution:
  1. Verify Ollama is running: ollama serve
  2. Check endpoint: http://localhost:11434
  3. Ensure network is active

Problem: "Model not found"
Solution:
  1. List available: ollama list
  2. Pull model: ollama pull bigdaddyg-fast:latest
  3. Wait for download to complete

Problem: "PowerShell execution policy error"
Solution:
  powershell -ExecutionPolicy Bypass -File "Launch-RawrXD-Agentic.ps1" -Terminal

Problem: "Cannot find type System.Windows.Forms"
Solution:
  Use -Terminal flag:
  .\Launch-RawrXD-Agentic.ps1 -Terminal

Problem: "Function not found"
Solution:
  1. Verify module loaded: Get-Module RawrXD-Agentic-Module
  2. Reload if needed: Import-Module .\RawrXD-Agentic-Module.psm1 -Force
  3. Check status: Get-RawrXDAgenticStatus

function Invoke-CodeReview {
    param([string]$FilePath)
    
    $code = Get-Content $FilePath -Raw
    $analysis = @{
        improvement = Invoke-RawrXDAgenticAnalysis -Code $code -AnalysisType 'improve'
        bugs = Invoke-RawrXDAgenticAnalysis -Code $code -AnalysisType 'debug'
        tests = Invoke-RawrXDAgenticAnalysis -Code $code -AnalysisType 'test'
    }
    return $analysis
}
