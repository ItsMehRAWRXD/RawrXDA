cmake_minimum_required(VERSION 3.20)
project(RawrXD-CLI VERSION 7.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Compiler flags
if(MSVC)
    add_compile_options(/O2 /W4 /EHsc)
else()
    add_compile_options(-O3 -Wall)
endif()

# Minimal CLI executable with Universal Generator
add_executable(rawrxd-cli
    src/main.cpp
    src/interactive_shell.cpp
    src/runtime_core.cpp
    src/universal_generator_service.cpp
    src/memory_core.cpp
    src/vsix_loader.cpp
    src/hot_patcher.cpp
    src/agentic_engine.cpp
    src/shared_context.cpp
)

# Include directories
target_include_directories(rawrxd-cli PRIVATE
    ${CMAKE_SOURCE_DIR}/src
    ${CMAKE_SOURCE_DIR}/src/engine
    ${CMAKE_SOURCE_DIR}/src/modules
)

# Link Threads
find_package(Threads REQUIRED)
target_link_libraries(rawrxd-cli PRIVATE Threads::Threads)

# Output
set_target_properties(rawrxd-cli PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)

# Optional: Add test executable for smoke testing
add_executable(rawrxd-smoke-test
    src/main.cpp
    src/interactive_shell.cpp
    src/runtime_core.cpp
    src/universal_generator_service.cpp
)

target_include_directories(rawrxd-smoke-test PRIVATE
    ${CMAKE_SOURCE_DIR}/src
    ${CMAKE_SOURCE_DIR}/src/engine
)

target_link_libraries(rawrxd-smoke-test PRIVATE Threads::Threads)

set_target_properties(rawrxd-smoke-test PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)
