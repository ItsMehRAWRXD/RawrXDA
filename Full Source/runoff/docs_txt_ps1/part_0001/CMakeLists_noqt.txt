cmake_minimum_required(VERSION 3.15)
project(RawrXD_NoQt)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Platform-specific settings for Windows
if(MSVC)
    add_compile_options(/W4 /permissive-)
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} /O2")
else()
    add_compile_options(-Wall -Wextra -Wpedantic)
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O3")
endif()

# Find required packages
find_package(Vulkan REQUIRED)

# Optional: nlohmann_json for JSON parsing
find_package(nlohmann_json 3.2.0 QUIET)

# Source files - NO Qt files
set(CORE_SOURCES
    src/qtapp/inference_engine_noqt.cpp
    src/qtapp/gguf_loader_noqt.cpp
    src/qtapp/bpe_tokenizer_noqt.cpp
    src/qtapp/vulkan_compute.cpp
    src/qtapp/transformer_inference.cpp
)

set(CORE_HEADERS
    src/qtapp/inference_engine_noqt.hpp
    src/qtapp/gguf_loader_noqt.hpp
    src/qtapp/bpe_tokenizer_noqt.hpp
    src/qtapp/vulkan_compute.h
    src/qtapp/transformer_inference.hpp
)

# API Server executable
add_executable(gguf_api_server
    src/gguf_api_server.cpp
    ${CORE_SOURCES}
)

target_include_directories(gguf_api_server PRIVATE
    src/qtapp
    ${VULKAN_INCLUDE_DIR}
)

target_link_libraries(gguf_api_server PRIVATE
    ${VULKAN_LIBRARIES}
    ws2_32  # Windows Sockets
    winmm   # Multimedia (for timing)
)

# Alternative: Simple API Server
add_executable(api_server_simple
    src/api_server_simple.cpp
    ${CORE_SOURCES}
)

target_include_directories(api_server_simple PRIVATE
    src/qtapp
    ${VULKAN_INCLUDE_DIR}
)

target_link_libraries(api_server_simple PRIVATE
    ${VULKAN_LIBRARIES}
    ws2_32
    winmm
)

# Tool Server executable
add_executable(tool_server
    src/tool_server.cpp
    ${CORE_SOURCES}
)

target_include_directories(tool_server PRIVATE
    src/qtapp
    ${VULKAN_INCLUDE_DIR}
)

target_link_libraries(tool_server PRIVATE
    ${VULKAN_LIBRARIES}
    ws2_32
    winmm
)

# Testing
enable_testing()

# Test: Tokenizer
add_executable(test_tokenizer
    tests/test_tokenizer.cpp
    src/qtapp/bpe_tokenizer_noqt.cpp
)

target_include_directories(test_tokenizer PRIVATE src/qtapp)
add_test(NAME TokenizerTest COMMAND test_tokenizer)

# Test: GGUF Loader
add_executable(test_gguf_loader
    tests/test_gguf_loader.cpp
    src/qtapp/gguf_loader_noqt.cpp
)

target_include_directories(test_gguf_loader PRIVATE src/qtapp)
add_test(NAME GGUFLoaderTest COMMAND test_gguf_loader)

# Output directories
set_target_properties(gguf_api_server PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)

set_target_properties(api_server_simple PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)

set_target_properties(tool_server PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin"
)

# Print build configuration
message(STATUS "=== RawrXD (Pure C++) Build Configuration ===")
message(STATUS "CMake version: ${CMAKE_VERSION}")
message(STATUS "C++ Standard: ${CMAKE_CXX_STANDARD}")
message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
message(STATUS "Vulkan found: YES")
message(STATUS "Vulkan version: ${VULKAN_VERSION}")
message(STATUS "Qt framework: NOT USED")
message(STATUS "Build targets:")
message(STATUS "  - gguf_api_server")
message(STATUS "  - api_server_simple")
message(STATUS "  - tool_server")
message(STATUS "============================================")
