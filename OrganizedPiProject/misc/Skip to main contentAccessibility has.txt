import java.security.MessageDigest;
import java.security.NoSuchAlgorithmException;
import java.util.*;

public class ExampleGenerator {
    
    static class Task {
        int id;
        String name;
        int difficulty;
        
        Task(int id, String name, int difficulty) {
            this.id = id;
            this.name = name;
            this.difficulty = difficulty;
        }
    }
    
    static final List<Task> TASKS = Arrays.asList(
        new Task(1, "hello_world", 1),
        new Task(2, "fibonacci", 2),
        new Task(3, "reverse_string", 2),
        new Task(4, "binary_search", 3),
        new Task(5, "merge_sort", 4)
    );
    
    static final Map<String, Map<String, String>> SNIPPETS = Map.of(
        "java", Map.of(
            "hello_world", "public class Hello { public static void main(String[] args) { System.out.println(\"Hello, world!\"); } }",
            "fibonacci", "public static int fib(int n) { if (n < 2) return n; return fib(n-1) + fib(n-2); }",
            "reverse_string", "public static String reverse(String s) { return new StringBuilder(s).reverse().toString(); }",
            "binary_search", "public static int binarySearch(int[] arr, int x) { int left = 0, right = arr.length-1; while (left <= right) { int mid = left + (right-left)/2; if (arr[mid] == x) return mid; if (arr[mid] < x) left = mid+1; else right = mid-1; } return -1; }",
            "merge_sort", "public static void mergeSort(int[] arr) { if (arr.length < 2) return; int mid = arr.length/2; int[] left = Arrays.copyOfRange(arr, 0, mid); int[] right = Arrays.copyOfRange(arr, mid, arr.length); mergeSort(left); mergeSort(right); merge(arr, left, right); }"
        )
    );
    
    public static List<Task> pickExamples(String lang, int n) {
        try {
            MessageDigest digest = MessageDigest.getInstance("SHA-256");
            byte[] hashBytes = digest.digest(lang.getBytes());
            
            long h = 0;
            for (int i = 0; i < 8; i++) {
                h = (h << 8) | (hashBytes[i] & 0xFF);
            }
            
            Set<Integer> seen = new HashSet<>();
            List<Task> result = new ArrayList<>();
            
            int[] indices = {
                (int)(Math.abs(h) % TASKS.size()),
                (int)(Math.abs(h / 7) % TASKS.size()),
                (int)(Math.abs(h / 13) % TASKS.size())
            };
            
            for (int idx : indices) {
                if (!seen.contains(idx)) {
                    seen.add(idx);
                    result.add(TASKS.get(idx));
                }
                if (result.size() == n) break;
            }
            
            return result;
        } catch (NoSuchAlgorithmException e) {
            throw new RuntimeException("SHA-256 not available", e);
        }
    }
    
    public static String toMarkdown(String lang, List<Task> tasks) {
        StringBuilder sb = new StringBuilder();
        sb.append("# ").append(lang.toUpperCase()).append(" examples\n\n");
        
        for (Task task : tasks) {
            String snippet = SNIPPETS.get(lang).get(task.name);
            sb.append("## ").append(task.name).append(" (difficulty ").append(task.difficulty).append(")\n");
            sb.append("
").append(lang).append("\n");
            sb.append(snippet).append("\n");
            sb.append("
\n\n");
        }
        
        return sb.toString();
    }
    
    public static String toJson(String lang, List<Task> tasks) {
        StringBuilder sb = new StringBuilder();
        sb.append("[\n");
        
        for (int i = 0; i < tasks.size(); i++) {
            Task task = tasks.get(i);
            String snippet = SNIPPETS.get(lang).get(task.name);
            sb.append("  {\n");
            sb.append("    \"id\": ").append(task.id).append(",\n");
            sb.append("    \"name\": \"").append(task.name).append("\",\n");
            sb.append("    \"difficulty\": ").append(task.difficulty).append(",\n");
            sb.append("    \"snippet\": \"").append(snippet.replace("\"", "\\\"")).append("\"\n");
            sb.append("  }");
            if (i < tasks.size() - 1) sb.append(",");
            sb.append("\n");
        }
        
        sb.append("]");
        return sb.toString();
    }
    
    public static void main(String[] args) {
        String lang = "java";
        List<Task> examples = pickExamples(lang, 3);
        
        System.out.println("=== MARKDOWN ===\n");
        System.out.println(toMarkdown(lang, examples));
        
        System.out.println("=== JSON ===\n");
        System.out.println(toJson(lang, examples));
    }
}

/**
 * Deterministically picks a number of examples based on the language.
 *
 * @param string $lang
 * @param int $n
 * @return array
 */
function pick_examples(string $lang, int $n, array $tasks): array
{
    // Use a hash of the language name to seed the selection.
    $hash = hash('sha256', $lang);
    $h = hexdec(substr($hash, 0, 16)); // Take the first 16 hex characters for a 64-bit integer.

    $seen = [];
    $result = [];

    $indices = [
        abs($h % count($tasks)),
        abs(intdiv($h, 7) % count($tasks)),
        abs(intdiv($h, 13) % count($tasks)),
    ];

    foreach ($indices as $idx) {
        if (!isset($seen[$idx])) {
            $seen[$idx] = true;
            $result[] = $tasks[$idx];
        }
        if (count($result) === $n) {
            break;
        }
    }

    return $result;
}

/**
 * Formats the given tasks into Markdown.
 *
 * @param string $lang
 * @param array $tasks
 * @param array $snippets
 * @return string
 */
function to_markdown(string $lang, array $tasks, array $snippets): string
{
    $output = "# " . strtoupper($lang) . " examples\n\n";

    foreach ($tasks as $task) {
        $snippet = $snippets[$lang][$task->name] ?? 'Snippet not found.';
        $output .= "## {$task->name} (difficulty {$task->difficulty})\n";
        $output .= "```{$lang}\n{$snippet}\n```\n\n";
    }

    return $output;
}

/**
 * Formats the given tasks into JSON.
 *
 * @param string $lang
 * @param array $tasks
 * @param array $snippets
 * @return string
 */
function to_json(string $lang, array $tasks, array $snippets): string
{
    $data = [];
    foreach ($tasks as $task) {
        $data[] = [
            'id' => $task->id,
            'name' => $task->name,
            'difficulty' => $task->difficulty,
            'snippet' => $snippets[$lang][$task->name] ?? 'Snippet not found.',
        ];
    }

    return json_encode($data, JSON_PRETTY_PRINT);
}

// --- Command-line argument handling ---
// Simple manual parsing since argparse isn't a native PHP feature.
$options = getopt('', ['lang:', 'format:']);
$lang = $options['lang'] ?? 'java';

$examples = pick_examples($lang, $n, $tasks);

if ($format === 'markdown') {
    echo to_markdown($lang, $examples, $snippets);
} elseif ($format === 'json') {
    echo to_json($lang, $examples, $snippets);
} else {
    echo "Unsupported format: {$format}\n";
    exit(1);
}

    ],

// --- Helper Functions ---
function pick(string $lang, int $n, array $tasks): array {
    $hash = hash('sha256', $lang, true);
    $s = unpack('J', substr($hash, 0, 8))[1];
    if ($s < 0) {
        $s = ~$s + 1; // PHP converts to float on 32-bit systems, so we manage negative manually.
    }

    $seen = [];
    $result = [];

    $task_count = count($tasks);
    $indices = [
        $s % $task_count,
        intdiv($s, 7) % $task_count,
        intdiv($s, 13) % $task_count,
    ];

    foreach ($indices as $idx) {
        if (!in_array($idx, $seen)) {
            $seen[] = $idx;
            $result[] = $tasks[$idx];
        }
    }

    $k = 1;
    while (count($result) < $n) {
        $new_idx = abs($s + 31 * $k) % $task_count;
        if (!in_array($new_idx, $seen)) {
            $seen[] = $new_idx;
            $result[] = $tasks[$new_idx];
        }
        $k++;
    }

    return array_slice($result, 0, min($n, count($result)));
}

function md(string $lang, array $tasks, array $snippets): string {
    $output = "# " . strtoupper($lang) . " examples\n\n";
    foreach ($tasks as $t) {
        $snippet = $snippets[$lang][$t->name] ?? '';
        $output .= "## {$t->name} (difficulty {$t->difficulty})\n```java\n{$snippet}\n```\n\n";
    }
    return $output;
}

function json(string $lang, array $tasks, array $snippets): string {
    $data = [];
    foreach ($tasks as $t) {
        $data[] = [
            "id" => $t->id,
            "name" => $t->name,
            "difficulty" => $t->difficulty,
            "snippet" => $snippets[$lang][$t->name] ?? '',
        ];
    }
    return json_encode($data, JSON_PRETTY_PRINT) . "\n";
}

function ask(string $prompt, string $url, ?string $auth): string {
    $ch = curl_init();
    $headers = ["Content-Type: application/json"];
    if ($auth !== null) {
        $headers[] = "Authorization: {$auth}";
    }

    $post_data = json_encode([
        "model" => "gpt-4o",
        "messages" => [
            ["role" => "system", "content" => "You are a helpful assistant."],
            ["role" => "user", "content" => $prompt]
        ],
        "temperature" => 0.2
    ]);

    curl_setopt($ch, CURLOPT_URL, $url);
    curl_setopt($ch, CURLOPT_POST, 1);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_HTTPHEADER, $headers);
    curl_setopt($ch, CURLOPT_POSTFIELDS, $post_data);

    $response = curl_exec($ch);
    $error = curl_error($ch);
    curl_close($ch);

    if ($error) {
        return "AI call failed ({$error}) – showing deterministic examples anyway.\n\n";
    }

    $json_response = json_decode($response, true);
    if (isset($json_response['choices'][0]['message']['content'])) {
        $ai_reply = $json_response['choices'][0]['message']['content'];
        return "AI reply:\n{$ai_reply}\n\n";
    } else {
        return "AI call failed (invalid response) – showing deterministic examples anyway.\n\n";
    }
}

$auth = null;

        $lang = $args[++$i];

The idea of Gemini working "like Google" is both true and misleading. Gemini is not a search engine in the traditional sense. It is deeply integrated into Google's search ecosystem to enhance it. 
Here is how Gemini is and is not like a search engine:
How Gemini is like a search engine
Information access: When a question is asked in Google's AI Overviews or the Gemini app, Gemini uses Google Search to retrieve information from across the web.
Conversational responses: Gemini can synthesize information from multiple sources into a comprehensive, conversational answer.
Deep research: Gemini has a "Deep Research" mode that can find answers to complex questions, similar to how intensive research might be done.
Up-to-date information: Gemini can be updated in near real-time, allowing it to provide fresh answers that go beyond the limitations of older AI models. 
How Gemini is different from a search engine
Role and function: A traditional search engine like Google Search is designed to index the web and present links to sources. Gemini acts as an AI assistant that can summarize, brainstorm, and plan with users, using the search results as its source material.
User interaction: A search engine responds to keywords and phrases with links, while Gemini can have a more fluid, interactive conversation with a user.
Result presentation: Google Search provides a direct list of web links. Gemini provides a summarized answer, which abstracts the user one step from the original sources.
Multimodality: Gemini is inherently multimodal, meaning it can process text, images, videos, and audio. This allows for new types of queries that go beyond text, like asking a question about a live video feed. 
In short, think of Gemini not as a replacement for Google Search, but as a layer of artificial intelligence that sits on top of Google's search infrastructure. This allows Google to provide more personalized, conversational, and comprehensive results than a traditional search alone.

import java.net.*;
import java.nio.charset.StandardCharsets;

    private static String ask(String prompt, String url, String auth) {
        try {
            URL apiEndpoint = new URL(url + "?key=" + URLEncoder.encode(auth, StandardCharsets.UTF_8));
            HttpURLConnection conn = (HttpURLConnection) apiEndpoint.openConnection();
            conn.setRequestMethod("POST");
            conn.setRequestProperty("Content-Type", "application/json");
            conn.setDoOutput(true);

            String jsonPayload = String.format(
                "{\"contents\":[{\"parts\":[{\"text\":\"%s\"}]}]}",
                prompt.replace("\\", "\\\\").replace("\"", "\\\"")
            );

            try (OutputStream os = conn.getOutputStream()) {
                byte[] input = jsonPayload.getBytes(StandardCharsets.UTF_8);
                os.write(input, 0, input.length);
            }

            int responseCode = conn.getResponseCode();
            if (responseCode >= 400) {
                return String.format("AI call failed (HTTP code: %d). Please check your API key and prompt.", responseCode);
            }

            try (BufferedReader br = new BufferedReader(new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8))) {
                StringBuilder response = new StringBuilder();
                String responseLine;
                while ((responseLine = br.readLine()) != null) {
                    response.append(responseLine.trim());
                }

                String jsonResponse = response.toString();
                // Simple parsing for the AI's text content
                int textStart = jsonResponse.indexOf("\"text\":\"") + 8;
                if (textStart > 8) {
                    int textEnd = jsonResponse.indexOf("\"", textStart);
                    if (textEnd != -1) {
                        return jsonResponse.substring(textStart, textEnd).replace("\\n", "\n").replace("\\\"","\"");
                    }
                }
                return "AI call failed (invalid or missing response).";
            }
        } catch (Exception e) {
            return "AI call failed (" + e.getMessage() + ").";
        }
    }

    private static String askAI(String prompt, String language, String url, String auth) {
        try {
            // Construct a more specific prompt for code generation.
            String fullPrompt = String.format(
                "Generate a clear, commented, and concise code example in %s for the following task: %s",
                language,
                prompt
            );

            // Append the API key to the URL
            URL apiEndpoint = new URL(url + "?key=" + URLEncoder.encode(auth, StandardCharsets.UTF_8));
            HttpURLConnection conn = (HttpURLConnection) apiEndpoint.openConnection();
            conn.setRequestMethod("POST");
            conn.setRequestProperty("Content-Type", "application/json");
            conn.setDoOutput(true);

            // Construct the JSON payload for the Gemini API
            String jsonPayload = String.format(
                "{\"contents\":[{\"parts\":[{\"text\":\"%s\"}]}]}",
                fullPrompt.replace("\\", "\\\\").replace("\"", "\\\"")
            );

            try (OutputStream os = conn.getOutputStream()) {
                byte[] input = jsonPayload.getBytes(StandardCharsets.UTF_8);
                os.write(input, 0, input.length);
            }

            if (responseCode >= 400) {

                    String errorResponse = br.lines().collect(Collectors.joining());
                    return String.format("AI call failed (HTTP code: %d). Details: %s", responseCode, errorResponse);

            try (BufferedReader br = new BufferedReader(new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8))) {

                }

<?php

 * Minimal AI CLI:
 *  - Only runs when --ask or --search is supplied
 *  - Talks to Google Generative Language API (Gemini)
 *  - Robust response parsing + clear errors
 */

function usage(): void {
    $u = <<<TXT
Usage:
  php ai_chat_only.php --ask "prompt" [--auth KEY|--env] [--model gemini-pro] [--system "text"] [--json]
  php ai_chat_only.php --search "query" [--auth KEY|--env] [--model gemini-pro] [--system "text"] [--json]
  php ai_chat_only.php --ask "..." --search "..." ...

Flags:
  --ask "text"        Ask the AI directly.
  --search "query"    Ask the AI to synthesize an answer as if from search results.
  --auth "KEY"        API key explicitly.
  --env               Read API key from GEMINI_API_KEY env var (ignored if --auth is present).
  --model "NAME"      Model name (default: gemini-pro). Examples: gemini-1.5-flash, gemini-1.5-pro.
  --api "URL"         Override API endpoint (default built from model).
  --system "text"     Optional system instruction.
  --timeout N         HTTP timeout in seconds (default: 30).
  --json              Print the raw JSON response instead of extracted text.

Notes:
  - Exits with code 1 for usage errors, 2 for HTTP/parse failures.
TXT;
    fwrite(STDERR, $u . PHP_EOL);
}

function build_url(string $model, ?string $api): string {
    if ($api !== null && $api !== '') return $api;
    // Default Gemini endpoint:
    return "https://generativelanguage.googleapis.com/v1beta/models/{$model}:generateContent";
}

function ai_call(string $prompt, ?string $system, string $url, string $api_key, int $timeout, bool $want_json) {
    $body = [
        "contents" => [[
            // role can be omitted; Gemini accepts plain contents
            "parts" => [["text" => $prompt]],
        ]],
    ];
    if ($system !== null && $system !== '') {
        $body["systemInstruction"] = [
            "parts" => [["text" => $system]],
        ];
    }

    $qs = strpos($url, '?') === false ? '?' : '&';
    $final_url = $url . $qs . "key=" . urlencode($api_key);

    $ch = curl_init($final_url);
    curl_setopt_array($ch, [
        CURLOPT_POST            => true,
        CURLOPT_RETURNTRANSFER  => true,
        CURLOPT_HTTPHEADER      => ["Content-Type: application/json"],
        CURLOPT_POSTFIELDS      => json_encode($body, JSON_UNESCAPED_UNICODE),
        CURLOPT_TIMEOUT         => $timeout,
    ]);
    $response = curl_exec($ch);
    $curl_err = curl_error($ch);
    $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);

    if ($curl_err || $response === false) {
        fwrite(STDERR, "AI call failed (network/cURL): {$curl_err}\n");
        exit(2);
    }
    if ($http_code >= 400) {
        $snippet = mb_substr($response, 0, 400);
        fwrite(STDERR, "AI call failed (HTTP {$http_code}). Server said:\n{$snippet}\n");
        exit(2);
    }

    $json = json_decode($response, true);
    if ($json === null) {
        fwrite(STDERR, "AI call failed: response was not valid JSON.\n");
        exit(2);
    }

    if ($want_json) {
        echo $response . PHP_EOL;
        return;
    }

    // Extract first text candidate safely
    $text = extract_text($json);
    if ($text === null) {
        // Some safety blocks return no text but include safetyFeedback
        $why = isset($json['promptFeedback']) ? json_encode($json['promptFeedback']) : 'no text candidates present';
        fwrite(STDERR, "AI call returned no text: {$why}\n");
        exit(2);
    }
    echo $text . PHP_EOL;
}

function extract_text(array $json): ?string {
    if (!isset($json['candidates']) || !is_array($json['candidates']) || count($json['candidates']) === 0) {
        return null;
    }
    $c = $json['candidates'][0] ?? null;
    if (!is_array($c)) return null;

    // Primary path: candidates[0].content.parts[*].text
    if (isset($c['content']['parts']) && is_array($c['content']['parts'])) {
        $parts = $c['content']['parts'];
        $texts = [];
        foreach ($parts as $p) {
            if (isset($p['text']) && is_string($p['text'])) {
                $texts[] = $p['text'];
            }
        }
        if ($texts) return implode("\n", $texts);
    }
    // Some variants may use 'text' directly on candidate (rare)
    if (isset($c['text']) && is_string($c['text'])) {
        return $c['text'];
    }
    return null;
}

    private static String buildUrl(String model, String api) {
        if (api != null && !api.isEmpty()) {
            return api;
        }
        return "https://generativelanguage.googleapis.com/v1beta/models/" + model + ":generateContent";

    private static void aiCall(String prompt, String system, String url, String apiKey, int timeout, boolean wantJson) {
        try {
            // Build the JSON body
            JsonNode body;
            if (system != null && !system.isEmpty()) {
                body = objectMapper.createObjectNode()
                    .putArray("contents").addObject().putArray("parts").addObject().put("text", prompt).end()
                    .end()
                    .set("systemInstruction", objectMapper.createObjectNode()
                        .putArray("parts").addObject().put("text", system).end()
                        .end());
            } else {
                body = objectMapper.createObjectNode()
                    .putArray("contents").addObject().putArray("parts").addObject().put("text", prompt).end()
                    .end();
            }
            String jsonPayload = objectMapper.writeValueAsString(body);

            // Append key to URL
            String finalUrl = url + "?key=" + URLEncoder.encode(apiKey, StandardCharsets.UTF_8);

            // Setup HTTP request
            HttpURLConnection conn = (HttpURLConnection) new URL(finalUrl).openConnection();
            conn.setRequestMethod("POST");
            conn.setRequestProperty("Content-Type", "application/json");
            conn.setDoOutput(true);
            conn.setConnectTimeout(timeout * 1000);
            conn.setReadTimeout(timeout * 1000);

            // Write payload
            try (OutputStream os = conn.getOutputStream()) {
                byte[] input = jsonPayload.getBytes(StandardCharsets.UTF_8);
                os.write(input, 0, input.length);
            }

            int responseCode = conn.getResponseCode();
            if (responseCode >= 400) {
                try (BufferedReader br = new BufferedReader(new InputStreamReader(conn.getErrorStream()))) {
                    String serverError = br.lines().collect(Collectors.joining());
                    System.err.printf("AI call failed (HTTP %d). Server said:%n%s%n", responseCode, serverError);
                }
                System.exit(2);
            }

            // Read response
            try (BufferedReader br = new BufferedReader(new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8))) {
                String response = br.lines().collect(Collectors.joining());
                if (wantJson) {
                    System.out.println(response);
                } else {
                    String text = extractText(objectMapper.readTree(response));
                    if (text == null) {
                        System.err.println("AI call returned no text candidates.");
                        System.exit(2);
                    }
                    System.out.println(text);
                }
            }
        } catch (Exception e) {
            System.err.printf("AI call failed (network/IO): %s%n", e.getMessage());
            System.exit(2);
        }
    }

    private static String extractText(JsonNode json) {
        if (json.has("candidates") && json.get("candidates").isArray() && !json.get("candidates").isEmpty()) {
            JsonNode candidate = json.get("candidates").get(0);
            if (candidate.has("content") && candidate.get("content").has("parts") && candidate.get("content").get("parts").isArray()) {
                StringBuilder texts = new StringBuilder();
                for (JsonNode part : candidate.get("content").get("parts")) {
                    if (part.has("text") && part.get("text").isTextual()) {
                        texts.append(part.get("text").asText());
                    }
                }
                return texts.toString();
            }
        }
        return null;
    }

What are other Java HTTP clients besides HttpURLConnection?
Java has several modern and robust HTTP client libraries that are generally preferred over HttpURLConnection for new development. These alternatives offer more features, better performance, and a more fluent and developer-friendly API. 
Here are some of the most popular alternatives:
java.net.http.HttpClient
This is the modern, built-in HTTP client that was officially released in Java 11, replacing HttpURLConnection.. 
Pros: It's part of the standard JDK, so no external dependencies are required. It supports HTTP/2, WebSockets, and a builder-driven, fluent API.
Cons: Not available on Java 8 and older versions.
Best for: Projects on Java 11 or higher that want a modern, lightweight, and dependency-free solution. 
Apache HttpClient
A long-standing and feature-rich open-source library that has been a standard in the Java ecosystem for many years. 
Pros: Highly configurable, battle-tested, and offers a vast array of features, including connection pooling, authentication, and request/response interception.
Cons: Can be seen as more heavyweight than other options.
Best for: Projects with complex or custom HTTP requirements, or for maintaining older applications. 
OkHttp
A modern, efficient, and well-maintained client from Square, often used in Android development but also popular for backend services. 
Pros: Excellent performance, supports HTTP/2 and connection pooling, and offers a simple, intuitive API.
Cons: It adds the Kotlin standard library as a dependency, which can be seen as bloat by some.
Best for: High-performance, production-ready applications that prioritize speed and efficiency. 
Retrofit
A type-safe HTTP client from Square that works on top of OkHttp. It uses Java annotations on an interface to define API endpoints. 
Pros: Provides a clean, declarative API that simplifies REST service consumption. It automatically handles request and response conversion using libraries like Jackson or Gson.
Cons: Best suited for consuming RESTful APIs, and less so for general-purpose HTTP.
Best for: Consuming RESTful APIs in a clean and type-safe manner. 
WebClient (Spring Framework)
Part of the Spring ecosystem, WebClient is a reactive, non-blocking HTTP client that offers a functional and fluent API. 
Pros: Ideal for asynchronous, high-concurrency applications within the Spring ecosystem.
Cons: Requires the Spring framework and has a steeper learning curve than synchronous clients.
Best for: Reactive applications using Spring WebFlux. 
Which client should you choose?
The "best" client depends heavily on your project's needs:
For simple, no-dependency use (Java 11+): java.net.http.HttpClient
For extensive features and configurability: Apache HttpClient
For high performance and a clean API: OkHttp
For a simplified, type-safe REST API client: Retrofit
For reactive applications in a Spring environment: WebClient

Give a specific use case comparing OkHttp and java.net.http.HttpClient
When comparing OkHttp and java.net.http.HttpClient, a specific use case that highlights their differences is building a high-performance, resilient REST API client in an environment where dependencies are not an issue. For this scenario, OkHttp is often the superior choice due to its advanced features. 
The use case: A high-performance API client
Imagine a financial services company building a microservice that needs to consume a third-party REST API to get real-time stock quotes. This service needs to be:
Fast: The microservice will handle thousands of concurrent requests, so network latency must be minimized.
Reliable: Network conditions are not always perfect, so the client must handle connection failures and temporary network hiccups gracefully.
Efficient: To minimize costs and resource usage, the client should use system resources efficiently.
Secure: Communication must be secure and use modern TLS protocols. 
Solution using OkHttp
OkHttp is a strong contender for this use case because it was built for production readiness, focusing on performance, reliability, and security out of the box.

// Maven dependency for OkHttp:

import okhttp3.*;
import java.io.IOException;

public class StockQuoteOkHttp {
    private static final OkHttpClient client = new OkHttpClient.Builder()
            .connectionPool(new ConnectionPool()) // Connection pooling enabled by default
            .retryOnConnectionFailure(true)      // Auto-recovers from connection failures
            .build();

    public static void main(String[] args) throws IOException {
        String symbol = "GOOG"; // Example stock symbol
        HttpUrl url = HttpUrl.get("https://api.example.com/stocks").newBuilder()
                .addQueryParameter("symbol", symbol)
                .build();

        Request request = new Request.Builder()
                .url(url)
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                throw new IOException("Unexpected code " + response);
            }
            System.out.println(response.body().string());
        }
    }
}

Why OkHttp excels in this scenario:
Performance: OkHttp offers advanced features like transparent GZIP compression and connection pooling, which minimize network overhead and latency, ideal for high-throughput scenarios.
Resilience: It automatically handles retries and recovers from transient network failures, which is critical for a service that must be robust against network instability.
Effortless setup: Its sensible, production-ready defaults mean that a basic client is fast, secure, and reliable with very little configuration.

import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.io.IOException;

public class StockQuoteHttpClient {
    private static final HttpClient client = HttpClient.newBuilder()
            .version(HttpClient.Version.HTTP_2) // HTTP/2 support, but connection handling is more basic
            .followRedirects(HttpClient.Redirect.NORMAL)
            .build();

    public static void main(String[] args) throws IOException, InterruptedException {
        String symbol = "GOOG";
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create("https://api.example.com/stocks?symbol=" + symbol))
                .build();

        HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());

        if (response.statusCode() != 200) {

Yes, there are several use cases where java.net.http.HttpClient is a better choice than OkHttp. The decision often comes down to a trade-off between avoiding third-party dependencies versus gaining specialized, out-of-the-box features. 
Use cases for preferring java.net.http.HttpClient
1. Small to medium-sized applications
For applications that don't have extremely demanding requirements for performance, resilience, or complex configurations, the built-in HttpClient is often sufficient. It provides all the necessary features for most REST API interactions without the overhead of external libraries. 
Example: A simple command-line tool that fetches data from a single API or a small server-side application that makes occasional HTTP calls.
2. Minimizing dependencies
Introducing external dependencies, especially in a small project or a microservice, can increase application size and complexity. The HttpClient is part of the Java Development Kit (JDK) starting from version 11, so it adds no new dependencies. 
Example: A project with strict rules about keeping its dependency graph small to minimize build times or reduce the risk of transitive dependency conflicts.
3. Enterprise environments with strict dependency policies
Many large organizations have strict policies regarding the use of third-party libraries. Some require a thorough security review and approval process for every new dependency. Using the standard library client avoids this process entirely. 
Example: A project developed for a government or financial institution that requires a tight lock on the software's components for security and compliance reasons.
4. Avoiding external library bloat
While OkHttp is known for being lean, it still brings in Okio as a transitive dependency, which some teams may want to avoid. For projects where every megabyte and dependency is scrutinized, the standard library is the cleanest option. 
Example: A small library or a lightweight serverless function where keeping the deployment package size to an absolute minimum is a priority.
5. Leveraging the modern JDK
The java.net.http.HttpClient API is modern, fluent, and designed to integrate seamlessly with other JDK features like CompletableFuture for asynchronous programming. This can lead to very readable and idiomatic Java code. 
Example: An application using reactive programming constructs or a microservice built with a framework that uses CompletableFuture extensively. 
6. Educational purposes

Example: A university course assignment or a personal project to practice core Java features.

public class SuperchargedCli {

    private static final ObjectMapper objectMapper = new ObjectMapper();

    private static final int EXIT_FAIL = 2;

    private static void usage() {
        String u = """

        System.err.println(u);
    }

    private static String buildUrl(String model, String api) {
        if (api != null && !api.isEmpty()) {
            return api;
        }
        return "https://generativelanguage.googleapis.com/v1beta/models/" + model + ":generateContent";
    }

    private static void aiCall(CliConfig config, String prompt) {
        try {
            JsonNode body = objectMapper.createObjectNode()
                .putArray("contents").addObject().putArray("parts").addObject().put("text", prompt).end().end();
            if (config.systemPrompt() != null && !config.systemPrompt().isEmpty()) {
                body = objectMapper.createObjectNode()
                    .putArray("contents").addObject().putArray("parts").addObject().put("text", prompt).end().end()
                    .set("systemInstruction", objectMapper.createObjectNode()
                        .putArray("parts").addObject().put("text", config.systemPrompt()).end().end());
            }
            
            String jsonPayload = objectMapper.writeValueAsString(body);
            String finalUrl = buildUrl(config.model(), config.api()) + "?key=" + URLEncoder.encode(config.apiKey(), StandardCharsets.UTF_8);

            HttpURLConnection conn = (HttpURLConnection) new URL(finalUrl).openConnection();
            conn.setRequestMethod("POST");
            conn.setRequestProperty("Content-Type", "application/json");
            conn.setDoOutput(true);
            conn.setConnectTimeout(config.timeout() * 1000);
            conn.setReadTimeout(config.timeout() * 1000);

            try (OutputStream os = conn.getOutputStream()) {
                os.write(jsonPayload.getBytes(StandardCharsets.UTF_8));
            }

            int responseCode = conn.getResponseCode();
            if (responseCode >= 400) {
                try (BufferedReader br = new BufferedReader(new InputStreamReader(conn.getErrorStream()))) {
                    String serverError = br.lines().collect(Collectors.joining());
                    System.err.printf("AI call failed (HTTP %d). Server said:%n%s%n", responseCode, serverError);
                }
                System.exit(EXIT_FAIL);
            }

            try (BufferedReader br = new BufferedReader(new InputStreamReader(conn.getInputStream(), StandardCharsets.UTF_8))) {
                String response = br.lines().collect(Collectors.joining());
                if (config.jsonOutput()) {
                    System.out.println(response);
                } else {
                    String text = extractText(objectMapper.readTree(response));
                    if (text == null) {
                        System.err.println("AI call returned no text candidates.");
                        System.exit(EXIT_FAIL);
                    }
                    System.out.println(text);
                }
            }
        } catch (Exception e) {
            System.err.printf("AI call failed (network/IO): %s%n", e.getMessage());
            System.exit(EXIT_FAIL);
        }
    }

    private static String extractText(JsonNode json) {
        if (json.has("candidates") && json.get("candidates").isArray() && !json.get("candidates").isEmpty()) {
            JsonNode candidate = json.get("candidates").get(0);
            if (candidate.has("content") && candidate.get("content").has("parts") && candidate.get("content").get("parts").isArray()) {
                StringBuilder texts = new StringBuilder();
                for (JsonNode part : candidate.get("content").get("parts")) {
                    if (part.has("text") && part.get("text").isTextual()) {
                        texts.append(part.get("text").asText());
                    }
                }
                return texts.toString();
            }
        }
        return null;
    }

    private static CliConfig parseArgs(String[] args) {
        CliConfig config = CliConfig.builder();
        
        for (int i = 0; i < args.length; i++) {
            String arg = args[i];
            switch (arg) {
                case "--ask":
                    if (i + 1 >= args.length) { usage(); System.exit(EXIT_USAGE); }
                    config.withAsk(args[++i]);
                    break;
                case "--search":
                    if (i + 1 >= args.length) { usage(); System.exit(EXIT_USAGE); }
                    config.withSearch(args[++i]);
                    break;
                case "--stdin":
                    config.withStdin();
                    break;
                case "--file":
                    if (i + 1 >= args.length) { usage(); System.exit(EXIT_USAGE); }
                    config.withFile(Path.of(args[++i]));
                    break;
                case "--batch":
                    if (i + 1 >= args.length) { usage(); System.exit(EXIT_USAGE); }
                    config.withBatch(Path.of(args[++i]));
                    break;
                case "--auth":
                    if (i + 1 >= args.length) { usage(); System.exit(EXIT_USAGE); }
                    config.withApiKey(args[++i]);
                    break;
                case "--env":
                    config.withEnvKey();
                    break;
                case "--model":
                    if (i + 1 >= args.length) { usage(); System.exit(EXIT_USAGE); }
                    config.withModel(args[++i]);
                    break;
                case "--json":
                    config.withJsonOutput();
                    break;
                case "--help":
                    usage();
                    System.exit(0);
                default:
                    System.err.println("Unknown flag: " + arg);
                    usage();
                    System.exit(EXIT_USAGE);
            }
        }
        return config;
    }

    public static void main(String[] args) {
        CliConfig config = parseArgs(args);
        
        if (config.inputSource() == null) {
            System.err.println("Please provide an input source (--ask, --search, etc.).");
            usage();
            System.exit(EXIT_USAGE);
        }
        if (config.apiKey() == null || config.apiKey().isEmpty()) {
            System.err.println("Missing API key. Use --auth KEY or --env with GEMINI_API_KEY set.");
            System.exit(EXIT_USAGE);
        }

        // The core logic would go here, processing based on config.inputSource()
        // For simplicity, we'll demonstrate a single call based on --ask or --search
        switch (config.inputSource()) {
            case ASK:
                aiCall(config, config.promptText());

What are other ways to manage concurrent requests in Java besides ThreadPoolExecutor?
There are several powerful and modern alternatives to ThreadPoolExecutor in Java for managing concurrent requests, each suited for different use cases. The choice depends on the specific requirements of your application, such as the nature of tasks (CPU-bound vs. I/O-bound), the need for explicit control, or the preference for a reactive programming style. 
1. CompletableFuture (Java 8+)
This is a popular choice for asynchronous programming that focuses on task composition. It allows you to define a pipeline of asynchronous steps, providing powerful methods for combining, chaining, and handling errors for multiple concurrent tasks. 
Best for: Executing a set of independent tasks concurrently and combining their results. This is particularly useful for I/O-bound tasks where threads might otherwise be idle while waiting for network requests to complete.
How it works: You can use CompletableFuture.runAsync() for tasks that don't return a value or CompletableFuture.supplyAsync() for tasks that do. You can use the thenCombine() or allOf() methods to wait for multiple futures to complete.
Example use case: In a web application, retrieving data from multiple external microservices concurrently and presenting a combined result to the user. 
2. Parallel Streams (Java 8+)
For data-intensive tasks that can be broken down and processed in parallel, Java Streams provide a simple way to enable multithreading. 
Best for: Processing large collections of data in parallel when the order of execution doesn't matter and tasks are stateless and independent.
How it works: Simply call the .parallelStream() method on a collection or .parallel() on an existing stream to have the runtime split the work across multiple threads using the common ForkJoinPool.
Example use case: Filtering, mapping, and reducing a very large list of records to compute an aggregate result. 
3. ForkJoinPool (Java 7+)
The ForkJoinPool is an implementation of ExecutorService specialized for "divide and conquer" algorithms. It uses a "work-stealing" algorithm to ensure all available threads are kept busy. 
Best for: Compute-intensive tasks that can be recursively broken down into smaller subtasks, such as highly parallel calculations and recursive algorithms.
How it works: Tasks are submitted as RecursiveAction (for no return value) or RecursiveTask (for a return value), which can then "fork" smaller subtasks. The "join" method is used to wait for subtasks to finish.
Example use case: Implementing a parallel sort or a complex image processing algorithm. 
4. Reactive Frameworks (e.g., RxJava, Project Reactor)
Reactive frameworks offer an asynchronous, event-driven approach to concurrency, often based on the Observer pattern. They handle streams of data and events in a non-blocking way, using a small number of threads to achieve high concurrency. 
Best for: I/O-bound applications that need to handle a large number of concurrent connections efficiently, like web servers or data streaming services.
How it works: You define a stream of events (Observable, Flux) and apply a series of operators to transform or consume the data. Schedulers manage which threads perform the work.
Example use case: A high-performance API gateway that needs to serve many simultaneous, long-lived requests without blocking threads. 
5. Virtual Threads (Java 19+, final in Java 21)
Introduced as Project Loom, virtual threads are lightweight, user-mode threads managed by the JVM, not the OS. This allows for a massive number of threads with very low overhead. 
Best for: Handling a large number of concurrent tasks, especially when those tasks are I/O-bound and would otherwise involve blocking. Virtual threads offer the simplicity of traditional thread-per-request concurrency without the heavy resource usage.
How it works: You can create them with Thread.ofVirtual() or use Executors.newVirtualThreadPerTaskExecutor(). The developer writes code as if it were a normal blocking thread, and the JVM automatically handles switching.
Example use case: A server application handling thousands of simultaneous requests, each handled by its own virtual thread, simplifying the code compared to asynchronous alternatives. 
6. Structured Concurrency (Java 19+, final in Java 21)
This is an API that simplifies concurrent programming by treating a group of related tasks as a single unit of work. It provides a more robust and observable way to manage the lifecycle of concurrent tasks. 
Best for: Managing a family of concurrent subtasks, especially when a task should fail if any of its subtasks fail. It helps prevent thread leaks and simplifies error handling and cancellation.
How it works: Use StructuredTaskScope within a try-with-resources block. You fork() subtasks and then join() them. The scope ensures that subtask lifetimes are confined to the block.
Example use case: A microservice that needs to fetch data from several other services and requires all of them to succeed before returning a response.

How does work-stealing in ForkJoinPool improve performance?
Work-stealing is a load-balancing algorithm used by the ForkJoinPool to maximize CPU utilization and improve performance. It is particularly effective for "divide and conquer" problems, where a large task is recursively broken down into smaller subtasks. 
How the work-stealing algorithm works
The process is based on a decentralized and collaborative model, avoiding a central task queue that can become a bottleneck under heavy load. 
Individual Task Deques: Each worker thread in the pool has its own double-ended queue (deque) of tasks.
LIFO for Owners: When a worker thread creates new subtasks (the "fork" step), it adds them to the front of its own deque and processes them in a Last-In, First-Out (LIFO) order. This improves cache performance, as the subtasks are likely to be related to the parent task and thus already in the processor's cache.
FIFO for Thieves: If a worker thread empties its own deque and becomes idle, it doesn't simply wait. Instead, it looks at the deques of other busy worker threads and "steals" a task from the back of their queue. This "victim" thread processes its own tasks from the front, minimizing contention with the stealing thread.
Dynamic Load Balancing: By continuously redistributing work from busy threads to idle ones, the algorithm ensures that all available CPU cores are kept busy as long as there is work to be done. This is the core of its efficiency.
Termination: The process continues until all tasks are executed and all worker threads become idle. 
Why it improves performance
Maximizes CPU Utilization: Work-stealing prevents threads from sitting idle, ensuring that all available CPU resources are used to their full potential. This leads to a higher throughput and faster completion of the overall task.
Dynamic Load Balancing: It is resilient to unbalanced workloads, where some tasks take longer to compute than others. The algorithm automatically balances the load, preventing bottlenecks that would occur in a fixed-scheduling system.
Minimizes Contention: The use of a deque and the back-to-front stealing strategy minimizes synchronization overhead. The owner of the deque and a potential thief are accessing opposite ends, which can happen with fewer lock conflicts.
Improves Cache Performance: By processing tasks in a LIFO order, the worker thread keeps its execution localized. The data required for these tasks is likely already in the CPU cache, reducing the need to fetch new data from memory.

What are the drawbacks of using ForkJoinPool for AI tasks?
While ForkJoinPool is a powerful tool for certain AI tasks, particularly compute-intensive, parallelizable ones, it comes with several significant drawbacks. These limitations can make it inefficient or even detrimental to performance, especially in scenarios involving I/O-bound operations, unbalanced tasks, or fine-grained parallelism. 
1. Inefficiency with I/O-bound tasks
The ForkJoinPool is designed for CPU-bound tasks. If a thread is blocked waiting for an I/O operation (e.g., network requests, disk reads), it cannot perform work-stealing and will effectively stall. This can lead to a situation where the common pool aggressively creates new threads to meet the parallelism level, potentially exhausting system resources and leading to an OutOfMemoryError. This is a critical consideration for AI tasks that involve fetching data from a database or an API, which are I/O-bound. 
2. High overhead for small tasks
The "fork" operation itself has a cost associated with task creation and management. If the subtasks are too small, the overhead of forking and joining can outweigh the benefits of parallelization. A single-threaded, sequential implementation might even be faster in these cases. For AI, this is a risk during tasks with very fine-grained parallelism, where the overhead of managing many tiny tasks could negate any performance gains. 
3. Suboptimal for non-recursive or non-divide-and-conquer tasks
The work-stealing algorithm is most efficient for tasks that follow a divide-and-conquer strategy. For a pool of independent tasks that are not recursively forked, ForkJoinPool offers little advantage over a standard ThreadPoolExecutor. Many AI tasks may not fit this mold perfectly, or the recursive nature may not be obvious, making a regular thread pool a simpler and more appropriate choice. 
4. Difficulty with blocking operations
The pool's performance can be severely degraded by blocking operations inside tasks. The internal ManagedBlocker mechanism attempts to mitigate this by creating extra threads, but this adds complexity and can still lead to performance bottlenecks. This is a concern for AI tasks that involve synchronization points, locks, or external resource contention. 
5. Potential for unbalanced workloads
While work-stealing is designed to handle unbalanced loads, it isn't a perfect solution. If a few tasks are significantly larger or more complex than others, they can still create bottlenecks. For example, in a genetic algorithm, the fitness evaluation for some individuals might be much more computationally expensive, and work-stealing might not perfectly balance the load, leading to some threads sitting idle while others are overloaded. 
6. Limited visibility and control
ForkJoinPool offers less visibility and control over thread management compared to other ExecutorService implementations. Monitoring the state of individual tasks or threads is more difficult. In complex AI systems where detailed performance tuning and debugging are critical, this lack of control can be a significant drawback. 
7. Complicated tuning
Determining the optimal "threshold" for when a recursive task should switch from forking to sequential execution is crucial but non-trivial. Setting this value too high or too low can negatively impact performance. The best threshold often requires experimentation and is highly dependent on the specific task and hardware. 
Alternatives for different AI use cases
For I/O-bound tasks: Use CompletableFuture with a standard ExecutorService (like newCachedThreadPool) or, on Java 21+, Executors.newVirtualThreadPerTaskExecutor().
For independent, fine-grained tasks: A standard ThreadPoolExecutor can be a simpler and more efficient solution, avoiding the overhead of the fork-join framework.
For reactive AI applications: Use reactive frameworks like Project Reactor or RxJava, which are designed for asynchronous and non-blocking I/O operations.

Let's look at how to choose the right thread pool size
Determining the right size for a thread pool is a critical performance tuning task. The ideal size is not a one-size-fits-all number, but depends primarily on the nature of the tasks being executed—whether they are CPU-bound or I/O-bound. 
Factors to consider
1. Task type (CPU-bound vs. I/O-bound)
CPU-bound tasks: These tasks consume significant CPU cycles and perform little to no I/O operations. Examples include heavy computations, image processing, or complex calculations.
Guideline: The ideal thread pool size is close to the number of available CPU cores. A common recommendation is N, where N is the number of cores, or N + 1 to account for potential page faults or minor delays. Increasing the thread count beyond this can lead to excessive context switching, which degrades performance.
I/O-bound tasks: These tasks spend most of their time waiting for I/O operations to complete, such as reading from a disk, waiting for a database query, or fetching data from a network API.
Guideline: A thread pool size larger than the number of CPU cores is beneficial, as it allows other threads to use the CPU while one is waiting for I/O. A common formula, proposed by Brian Goetz, is Number of threads = Number of Available Cores * (1 + Wait time / Service time). The ratio Wait time / Service time is known as the "blocking coefficient." 
2. Number of available CPU cores
You can determine the number of available cores programmatically in Java using Runtime.getRuntime().availableProcessors(). It's crucial to know the hardware capabilities of the machine running your application. 
3. Wait time and service time
For I/O-bound tasks, accurately estimating the ratio of time spent waiting on external resources (Wait time) to the time spent on active computation (Service time) is key. 
Wait time: The average time a thread spends blocked, for instance, waiting for a network response.
Service time: The average time a thread spends actively computing. 
4. System and resource limits
You must consider constraints beyond CPU cores. 
External service limits: Some external APIs or database connection pools have their own concurrency limits. Creating more threads than these external services can handle may lead to resource contention.
Memory: Each thread consumes memory for its stack. An excessively large thread pool can exhaust your application's memory, especially if tasks use a large amount of memory. 
5. Monitoring and tuning
Ultimately, calculating a perfect thread pool size upfront is difficult due to the many variables involved. The best practice is to: 
Start with a reasonable heuristic based on your task type.
Make the pool size a configurable parameter so it can be adjusted without code changes.
Monitor your application under realistic load conditions, observing metrics like CPU utilization, latency, and throughput.
Tune the pool size based on your monitoring results, increasing it to improve throughput until performance starts to degrade. 
Practical implementation
For applications with mixed workloads (some CPU-bound, some I/O-bound), a robust approach is to use multiple thread pools, with each pool sized appropriately for its specific task type. 
Final thought on virtual threads
For I/O-bound tasks on Java 21+, virtual threads largely eliminate the need to manually tune thread pool sizes. The Executors.newVirtualThreadPerTaskExecutor() dynamically creates a new lightweight virtual thread for each task, allowing for massive concurrency with minimal overhead and without blocking OS threads. This approach simplifies concurrency management for many applications.

function load_json_file(string $path): ?array {
    if (!is_file($path)) return null;
    $s = file_get_contents($path);
    if ($s === false) return null;
    $j = json_decode($s, true);
    return is_array($j) ? $j : null;
}

function save_json_file(string $path, array $data): void {
    $dir = dirname($path);
    if (!is_dir($dir)) @mkdir($dir, 0777, true);
    file_put_contents($path, json_encode($data, JSON_PRETTY_PRINT|JSON_UNESCAPED_UNICODE));
}

function extract_text_from_response(array $json): ?string {
    if (!isset($json['candidates']) || !is_array($json['candidates']) || count($json['candidates']) === 0) return null;
    $c = $json['candidates'][0];
    // Common path: candidates[0].content.parts[*].text
    $out = [];
    if (isset($c['content']['parts']) && is_array($c['content']['parts'])) {
        foreach ($c['content']['parts'] as $p) {
            if (isset($p['text']) && is_string($p['text'])) $out[] = $p['text'];
        }
    }
    if ($out) return implode("\n", $out);
    // fallback
    if (isset($c['text']) && is_string($c['text'])) return $c['text'];
    return null;
}

function backoff_sleep_ms(int $ms): void {
    usleep($ms * 1000);
}

function ai_call(array $payload, string $url, string $api_key, int $timeout, ?string $proxy, int $retries, int $backoff_base_ms, bool $verbose): string {
    $qs = strpos($url, '?') === false ? '?' : '&';
    $final_url = $url . $qs . "key=" . urlencode($api_key);
    $attempt = 0;
    $last_err = "";

    while (true) {
        $attempt++;
        $ch = curl_init($final_url);
        $headers = ["Content-Type: application/json"];
        $opts = [
            CURLOPT_POST           => true,
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_HTTPHEADER     => $headers,
            CURLOPT_POSTFIELDS     => json_encode($payload, JSON_UNESCAPED_UNICODE),
            CURLOPT_TIMEOUT        => $timeout,
        ];
        if ($proxy) $opts[CURLOPT_PROXY] = $proxy;
        curl_setopt_array($ch, $opts);

        $resp = curl_exec($ch);
        $curl_err = curl_error($ch);
        $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
        curl_close($ch);

        if ($curl_err || $resp === false) {
            $last_err = "Network/cURL error: {$curl_err}";
            if ($verbose) fwrite(STDERR, "[attempt {$attempt}] {$last_err}\n");
            if ($attempt <= $retries) {
                $sleep = $backoff_base_ms * (2 ** ($attempt - 1));
                backoff_sleep_ms($sleep);
                continue;
            }
            fwrite(STDERR, $last_err . "\n");
            exit(2);
        }

        if ($http_code >= 500 || $http_code == 429) {
            $last_err = "HTTP {$http_code}: transient/server rate-limit.\n" . mb_substr($resp, 0, 400);
            if ($verbose) fwrite(STDERR, "[attempt {$attempt}] {$last_err}\n");
            if ($attempt <= $retries) {
                $sleep = $backoff_base_ms * (2 ** ($attempt - 1));
                backoff_sleep_ms($sleep);
                continue;
            }
            fwrite(STDERR, $last_err . "\n");
            exit(2);
        }

        if ($http_code >= 400) {
            $snippet = mb_substr($resp, 0, 800);
            fwrite(STDERR, "HTTP {$http_code} error:\n{$snippet}\n");
            exit(2);
        }

        return $resp;
    }
}
