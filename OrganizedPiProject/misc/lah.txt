@Bean
public WebSocketHandler burrrHandler() {
    return session -> session.receive()
            .flatMap(msg -> {
                ByteBuffer buf = msg.getPayload();
                switch (buf.get(0)) {           // 1-byte opcode
                    case 0x01 -> handleZip(session, buf);
                    case 0x02 -> handleBuild(session, buf);
                    default   -> Mono.empty();
                }
            });
}

private Mono<Void> handleZip(WebSocketSession s, ByteBuffer buf) {
    Path work = Files.createTempDirectory("burrr");
    try (var zis = new ZipInputStream(new ByteArrayInputStream(buf.array()))) {
        zis.unzipTo(work);
    }
    s.getAttributes().put("work", work);
    return s.send(Mono.just(s.textMessage("{\"type\":\"OK\"}")));
}

private Mono<Void> handleBuild(WebSocketSession s, ByteBuffer buf) {
    Path work = (Path) s.getAttributes().get("work");
    return Flux.using(
            () -> new ProcessBuilder("./gradlew", "assembleDebug")
                    .directory(work.toFile()).start(),
            proc -> Flux.merge(
                    byteReader(proc.getInputStream()).map(l -> log("INFO", l)),
                    byteReader(proc.getErrorStream()).map(l -> log("ERROR", l))),
            Process::destroy)
            .concatWith(artifact(work))
            .map(s::textMessage)
            .as(s::send);
}

private String log(String lvl, String line) {
    return "{\"type\":\"LOG\",\"level\":\""+lvl+"\",\"chunk\":"+Json.encode(line)+"}";
}

private String artifact(Path work) {
    Path apk = work.resolve("app/build/outputs/apk/debug/app-debug.apk");
    if (Files.exists(apk)) {
        String sha = sha256(apk);
        return "{\"type\":\"ARTIFACT\",\"size\":"+Files.size(apk)+",\"sha256\":\""+sha+"\"}";
    }
    return "{\"type\":\"DONE\"}";
}

private void refreshFileTree() {
    DefaultMutableTreeNode root = (DefaultMutableTreeNode) treeModel.getRoot();
    root.removeAllChildren();
    addDirectoryToTree(root, new File("project"));
    treeModel.reload();
    expandTree();
}

private void addDirectoryToTree(DefaultMutableTreeNode parent, File dir) {
    File[] files = dir.listFiles();
    if (files != null) {
        Arrays.sort(files, (a, b) -> {
            if (a.isDirectory() && !b.isDirectory()) return -1;
            if (!a.isDirectory() && b.isDirectory()) return 1;
            return a.getName().compareTo(b.getName());
        });
        
        for (File file : files) {
            DefaultMutableTreeNode node = new DefaultMutableTreeNode(file.getName());
            parent.add(node);
            if (file.isDirectory()) {
                addDirectoryToTree(node, file);
            }
        }
    }
}

private void expandTree() {
    for (int i = 0; i < fileTree.getRowCount(); i++) {
        fileTree.expandRow(i);
    }
}

private void autoFixAndSave() {
    JTextArea currentEditor = getCurrentEditor();
    if (currentEditor != null) {
        String code = currentEditor.getText();
        executor.submit(() -> {
            // Get fixes from both Rawr and Amazon Q
            String rawrFixes = rawrCopilot.suggestFixes(code);
            String amazonQFixes = amazonQ.suggestFixes(code);
            
            SwingUtilities.invokeLater(() -> {
                // Apply fixes automatically without updating editor display
                String fixedCode = applyFixes(code, rawrFixes, amazonQFixes);
                saveCurrentFileWithContent(fixedCode);
                log("Auto-fix applied and saved silently");
                showInChat("Combined AI", "Auto-fixes applied:\nRawr: " + rawrFixes + "\nAmazon Q: " + amazonQFixes);
            });
        });
    }
}

private String applyFixes(String originalCode, String rawrFixes, String amazonQFixes) {
    // Simple fix application - in real implementation, this would be more sophisticated
    String fixedCode = originalCode;
    
    // Apply security fixes from Rawr
    if (rawrFixes.contains("SQL injection")) {
        fixedCode = fixedCode.replaceAll("Statement", "PreparedStatement");
    }
    
    // Apply code quality fixes from Amazon Q
    if (amazonQFixes.contains("null check")) {
        fixedCode = "// Auto-added null checks\n" + fixedCode;
    }
    
    return fixedCode;
}

private JTextArea getCurrentEditor() {
    int selectedTab = editorTabs.getSelectedIndex();
    if (selectedTab >= 0) {
        String fileName = editorTabs.getTitleAt(selectedTab);
        return openFiles.get(fileName);
    }
    return null;
}

private void openSelectedFile() {
    TreePath path = fileTree.getSelectionPath();
    if (path != null) {
        String filePath = getFullPath(path);
        File file = new File(filePath);
        if (file.isFile()) {
            openFile(file);
        }
    }
}

private String getFullPath(TreePath path) {
    StringBuilder sb = new StringBuilder();
    for (int i = 1; i < path.getPathCount(); i++) {
        if (sb.length() > 0) sb.append("/");
        sb.append(path.getPathComponent(i).toString());
    }
    return "project/" + sb.toString();
}

private void openFile(File file) {
    String fileName = file.getName();
    if (openFiles.containsKey(fileName)) {
        for (int i = 0; i < editorTabs.getTabCount(); i++) {
            if (editorTabs.getTitleAt(i).equals(fileName)) {
                editorTabs.setSelectedIndex(i);
                return;
            }
        }
    }
    
    try {
        String content = new String(Files.readAllBytes(file.toPath()));
        JTextArea editor = new JTextArea(content);
        editor.setFont(new Font("Consolas", Font.PLAIN, 14));
        
        JScrollPane scrollPane = new JScrollPane(editor);
        editorTabs.addTab(fileName, scrollPane);
        editorTabs.setSelectedComponent(scrollPane);
        
        openFiles.put(fileName, editor);
        
    } catch (Exception e) {
        log("Error opening file: " + e.getMessage());
    }
}

private void createNewFile() {
    String fileName = JOptionPane.showInputDialog(this, "Enter file name:");
    if (fileName != null && !fileName.trim().isEmpty()) {
        try {
            Path filePath = Paths.get("project/src/" + fileName);
            Files.write(filePath, "// New file created by Rawr + Amazon Q IDE\n".getBytes());
            refreshFileTree();
            openFile(filePath.toFile());
        } catch (Exception e) {
            log("Error creating file: " + e.getMessage());
        }
    }
}

private void saveAll() {
    for (Map.Entry<String, JTextArea> entry : openFiles.entrySet()) {
        saveFile(entry.getKey(), entry.getValue().getText());
    }
    log("All files saved");
}

private void autoSaveAll() {
    saveAll();
}

private void saveFile(String fileName, String content) {
    try {
        Path filePath = findFile(Paths.get("project"), fileName);
        if (filePath != null) {
            Files.write(filePath, content.getBytes());
        }
    } catch (Exception e) {
        log("Error saving file " + fileName + ": " + e.getMessage());
    }
}

private Path findFile(Path dir, String fileName) throws IOException {
    return Files.walk(dir)
            .filter(path -> path.getFileName().toString().equals(fileName))
            .findFirst()
            .orElse(null);
}

private void log(String message) {
    SwingUtilities.invokeLater(() -> {
        outputArea.append("[" + new java.text.SimpleDateFormat("HH:mm:ss").format(new Date()) + "] " + message + "\n");
        outputArea.setCaretPosition(outputArea.getDocument().getLength());
    });
}

        }
    }
    
    private void processAIRequest(String request) {
        executor.submit(() -> {
            String response = generateAIResponse(request);
            SwingUtilities.invokeLater(() -> addMessage(name, response));
        });
    }
    
    private String generateAIResponse(String request) {
        try {
            Thread.sleep(800);
            
            switch (type) {
                case "rawr-security":
                    if (request.toLowerCase().contains("encrypt")) {
                        return "Rawr: I can encrypt your files with AES-256, RSA-2048, or Blowfish. Which would you prefer?";
                    } else if (request.toLowerCase().contains("security")) {
                        return "Rawr: I've analyzed your code for security vulnerabilities. Check the security analysis results.";
                    }
                    return "Rawr: I'm your security copilot. I can encrypt, analyze, and secure your code.";
                    
                case "amazon-q-dev":
                    if (request.toLowerCase().contains("test")) {
                        return "Amazon Q: I can generate comprehensive unit tests for your code. Would you like me to create them?";
                    } else if (request.toLowerCase().contains("review")) {
                        return "Amazon Q: I've reviewed your code for best practices and potential improvements.";
                    }
                    return "Amazon Q: I'm here to help with code development, testing, and reviews.";
                    
                case "amazon-q-chat":
                    return "Amazon Q: I can explain any part of your code in detail. Just select the code you want me to explain.";
                    
                case "combined-ai":
                    return "Combined AI: Both Rawr and Amazon Q are working together to provide security analysis and code improvements.";
                    
                default:
                    return "AI: How can I help you today?";
            }
        } catch (InterruptedException e) {
            return "AI: Response generation interrupted";
        }
    }
    
    public void addMessage(String sender, String msg) {
        SwingUtilities.invokeLater(() -> {
            chat.append(String.format("[%s] %s: %s\n", 
                new java.text.SimpleDateFormat("HH:mm:ss").format(new Date()), sender, msg));
            chat.setCaretPosition(chat.getDocument().getLength());
        });
    }

private void addDirectoryToTree(DefaultMutableTreeNode parent, File dir) {
    File[] files = dir.listFiles();
    if (files != null) {
        Arrays.sort(files, (a, b) -> {
            if (a.isDirectory() && !b.isDirectory()) return -1;
            if (!a.isDirectory() && b.isDirectory()) return 1;
            return a.getName().compareTo(b.getName());
        });
        
        for (File file : files) {
            DefaultMutableTreeNode node = new DefaultMutableTreeNode(file.getName());
            parent.add(node);
            if (file.isDirectory()) {
                addDirectoryToTree(node, file);
            }
        }
    }
}

private void expandTree() {
    for (int i = 0; i < fileTree.getRowCount(); i++) {
        fileTree.expandRow(i);
    }
}

private JTextArea getCurrentEditor() {
    int selectedTab = editorTabs.getSelectedIndex();
    if (selectedTab >= 0) {
        String fileName = editorTabs.getTitleAt(selectedTab);
        return openFiles.get(fileName);
    }
    return null;
}

private void openSelectedFile() {
    TreePath path = fileTree.getSelectionPath();
    if (path != null) {
        String filePath = getFullPath(path);
        File file = new File(filePath);
        if (file.isFile()) {
            openFile(file);
        }
    }
}

private String getFullPath(TreePath path) {
    StringBuilder sb = new StringBuilder();
    for (int i = 1; i < path.getPathCount(); i++) {
        if (sb.length() > 0) sb.append("/");
        sb.append(path.getPathComponent(i).toString());
    }
    return "mcp-project/" + sb.toString();
}

private void openFile(File file) {
    String fileName = file.getName();
    if (openFiles.containsKey(fileName)) {
        for (int i = 0; i < editorTabs.getTabCount(); i++) {
            if (editorTabs.getTitleAt(i).equals(fileName)) {
                editorTabs.setSelectedIndex(i);
                return;
            }
        }
    }
    
    try {
        String content = new String(Files.readAllBytes(file.toPath()));
        JTextArea editor = new JTextArea(content);
        editor.setFont(new Font("Monospaced", Font.PLAIN, 14));
        
        JScrollPane scrollPane = new JScrollPane(editor);
        editorTabs.addTab(fileName, scrollPane);
        editorTabs.setSelectedComponent(scrollPane);
        
        openFiles.put(fileName, editor);
        
    } catch (Exception e) {
        log("Error opening file: " + e.getMessage());
    }
}

private void createNewFile() {
    String fileName = JOptionPane.showInputDialog(this, "Enter file name:");
    if (fileName != null && !fileName.trim().isEmpty()) {
        try {
            Path filePath = Paths.get("mcp-project/src/" + fileName);
            Files.write(filePath, "// MCP-Enhanced file\n".getBytes());
            refreshFileTree();
            openFile(filePath.toFile());
        } catch (Exception e) {
            log("Error creating file: " + e.getMessage());
        }
    }
}

private void saveAll() {
    for (Map.Entry<String, JTextArea> entry : openFiles.entrySet()) {
        saveFile(entry.getKey(), entry.getValue().getText());
    }
    log("All files saved with MCP context");
}

private void autoSaveAll() {
    saveAll();
}

private void saveFile(String fileName, String content) {
    try {
        Path filePath = findFile(Paths.get("mcp-project"), fileName);
        if (filePath != null) {
            Files.write(filePath, content.getBytes());
            mcpServer.notifyFileChange(fileName, content);
        }
    } catch (Exception e) {
        log("Error saving file " + fileName + ": " + e.getMessage());
    }
}

private Path findFile(Path dir, String fileName) throws IOException {
    return Files.walk(dir)
            .filter(path -> path.getFileName().toString().equals(fileName))
            .findFirst()
            .orElse(null);
}

private void updateIDE() {
    String updateCode = JOptionPane.showInputDialog(this, 
        "Enter MCP-enhanced update code:");
    
    if (updateCode != null && !updateCode.trim().isEmpty()) {
        try {
            String mcpValidation = mcpServer.validateUpdate(updateCode);
            if (mcpValidation.equals("VALID")) {
                String currentSource = getCurrentIDESource();
                String updatedSource = insertUpdateCode(currentSource, updateCode);
                
                Files.write(Paths.get("MCPIntegratedIDE.java"), updatedSource.getBytes());
                
                Process compile = Runtime.getRuntime().exec("javac MCPIntegratedIDE.java");
                compile.waitFor();
                
                if (compile.exitValue() == 0) {
                    log("MCP-validated IDE update successful!");
                    askAllAIsMCP("IDE updated with MCP validation: " + updateCode);
                } else {
                    log("Compilation failed. Update rejected.");
                }
            } else {
                log("MCP validation failed: " + mcpValidation);
            }
            
        } catch (Exception e) {
            log("Error updating IDE: " + e.getMessage());
        }
    }
}

private String getCurrentIDESource() throws IOException {
    return new String(Files.readAllBytes(Paths.get("MCPIntegratedIDE.java")));
}

private String insertUpdateCode(String source, String updateCode) {
    int lastBrace = source.lastIndexOf("}");
    return source.substring(0, lastBrace) + 
           "\n    // MCP-VALIDATED UPDATE\n    " + 
           updateCode + "\n" + 
           source.substring(lastBrace);
}

private void log(String message) {
    SwingUtilities.invokeLater(() -> {
        outputArea.append("[" + new java.text.SimpleDateFormat("HH:mm:ss").format(new Date()) + "] " + message + "\n");
        outputArea.setCaretPosition(outputArea.getDocument().getLength());
    });
}

private String filterEmojis(String text) {
    if (EMOJI_PATTERN.matcher(text).find()) {
        return "[EMOJI FILTERED] " + EMOJI_PATTERN.matcher(text).replaceAll("");
    }
    return text;
}

// MCP Server Implementation
class MCPServer {
    private boolean running = false;
    private Map<String, String> tools = new HashMap<>();
    private Map<String, String> resources = new HashMap<>();
    
    public void start() {
        running = true;
        initializeTools();
        log("MCP Server started with tools and resources");
    }
    
    private void initializeTools() {
        tools.put("code_analyzer", "Analyzes code structure and quality");
        tools.put("file_manager", "Manages project files and directories");
        tools.put("git_helper", "Git operations and version control");
        tools.put("compiler_assistant", "Compilation and build assistance");
        tools.put("test_generator", "Generates unit tests");
    }
    
    public void connectClient(String clientUrl) {
        log("MCP Client connected: " + clientUrl);
    }
    
    public String[] listTools() {
        return tools.keySet().toArray(new String[0]);
    }
    
    public String executeTool(String toolName, String params) {
        if (tools.containsKey(toolName)) {
            return "MCP Tool '" + toolName + "' executed with params: " + params;
        }
        return "Tool not found: " + toolName;
    }
    
    public String getCodeContext(String code) {
        return "MCP Context: Java code, " + code.split("\n").length + " lines, contains main method: " + code.contains("main");
    }
    
    public String getAvailableTools() {
        return String.join(", ", tools.keySet());
    }
    
    public void notifyFileChange(String fileName, String content) {
        log("MCP: File change notification - " + fileName);
    }
    
    public String validateUpdate(String code) {
        if (code.contains("System.exit") || code.contains("Runtime.exec")) {
            return "INVALID: Potentially dangerous code";
        }
        return "VALID";
    }
}

// MCP-Enhanced AI Chat
class MCPAIChat extends JPanel {
    private JTextArea chat;
    private JTextField input;
    private String name, model;
    private MCPServer mcpServer;
    
    MCPAIChat(String name, String model, MCPServer mcpServer) {
        this.name = name;
        this.model = model;
        this.mcpServer = mcpServer;
        
        setLayout(new BorderLayout());
        setBorder(BorderFactory.createTitledBorder(name + " - MCP Enhanced - NO EMOJIS"));
        
        chat = new JTextArea();
        chat.setEditable(false);
        chat.setFont(new Font("Segoe UI", Font.PLAIN, 11));
        
        input = new JTextField();
        input.addActionListener(e -> sendMessage());
        
        JButton sendBtn = new JButton("Send");
        sendBtn.addActionListener(e -> sendMessage());
        
        JPanel inputPanel = new JPanel(new BorderLayout());
        inputPanel.add(input, BorderLayout.CENTER);
        inputPanel.add(sendBtn, BorderLayout.EAST);
        
        add(new JScrollPane(chat), BorderLayout.CENTER);
        add(inputPanel, BorderLayout.SOUTH);
        
        addMessage("System", "MCP-Enhanced " + name + " ready - Protocol active");
    }
    
    private void sendMessage() {
        String msg = input.getText().trim();
        if (!msg.isEmpty()) {
            addMessage("You", msg);
            input.setText("");
            askAIWithMCP(msg);
        }
    }
    
    void askAIWithMCP(String question) {
        addMessage("System", "Querying " + name + " via MCP...");
        
        new Thread(() -> {
            try {
                String mcpContext = mcpServer.getCodeContext(question);
                String response = callMCPAI(question, mcpContext);
                SwingUtilities.invokeLater(() -> addMessage(name, response));
            } catch (Exception e) {
                SwingUtilities.invokeLater(() -> addMessage("Error", "MCP AI request failed: " + e.getMessage()));
            }
        }).start();
    }
    
    void receiveMCPResult(String toolName, String result) {
        addMessage("MCP Tool", toolName + " result: " + result);
    }
    
    private String callMCPAI(String prompt, String mcpContext) throws Exception {
        Thread.sleep(800 + (int)(Math.random() * 1200));
        
        switch (name) {
            case "MCP-GPT":
                return "MCP-GPT Analysis with context [" + mcpContext + "]: " + generateMCPResponse(prompt);
            case "MCP-Claude":
                return "MCP-Claude Response with tools: " + generateClaudeMCPResponse(prompt);
            case "MCP-Q":
                return "MCP-Amazon Q with AWS context: " + generateQMCPResponse(prompt);
            case "MCP-Multi":
                return "MCP Multi-Agent: All agents processed via MCP protocol. Consensus reached.";
            default:
                return "MCP Processing: " + prompt;
        }
    }
    
    private String generateMCPResponse(String prompt) {
        if (prompt.contains("code")) {
            return "Code analysis complete via MCP tools. Structure validated, no critical issues found.";
        }
        return "MCP-enhanced response ready. Tools available for deeper analysis.";
    }
    
    private String generateClaudeMCPResponse(String prompt) {
        return "MCP protocol enables enhanced reasoning. Context preserved across tool calls.";
    }
    
    private String generateQMCPResponse(String prompt) {
        return "AWS MCP integration active. Cloud resources and services accessible via protocol.";
    }
    
    void addMessage(String sender, String msg) {
        SwingUtilities.invokeLater(() -> {
            String filtered = filterEmojis(msg);
            chat.append(String.format("[%s] %s: %s\n", 
                new java.text.SimpleDateFormat("HH:mm:ss").format(new Date()), sender, filtered));
            chat.setCaretPosition(chat.getDocument().getLength());
        });
    }
}

private void addDirectoryToTree(DefaultMutableTreeNode parent, File dir) {
    File[] files = dir.listFiles();
    if (files != null) {
        Arrays.sort(files, (a, b) -> {
            if (a.isDirectory() && !b.isDirectory()) return -1;
            if (!a.isDirectory() && b.isDirectory()) return 1;
            return a.getName().compareTo(b.getName());
        });
        
        for (File file : files) {
            DefaultMutableTreeNode node = new DefaultMutableTreeNode(file.getName());
            parent.add(node);
            if (file.isDirectory()) {
                addDirectoryToTree(node, file);
            }
        }
    }
}

}

private void openFile(File file) {
    String fileName = file.getName();
    if (openFiles.containsKey(fileName)) {
        for (int i = 0; i < editorTabs.getTabCount(); i++) {
            if (editorTabs.getTitleAt(i).equals(fileName)) {
                editorTabs.setSelectedIndex(i);
                return;
            }
        }
    }
    
    try {
        String content = new String(Files.readAllBytes(file.toPath()));
        JTextArea editor = new JTextArea(content);
        editor.setFont(new Font("Monospaced", Font.PLAIN, 14));
        
        JScrollPane scrollPane = new JScrollPane(editor);
        editorTabs.addTab(fileName, scrollPane);
        editorTabs.setSelectedComponent(scrollPane);
        
        openFiles.put(fileName, editor);
        
    } catch (Exception e) {
        log("Error opening file: " + e.getMessage());
    }
}

private void updateIDE() {
    String updateCode = JOptionPane.showInputDialog(this, 
        "Enter MCP-enhanced update code:");
    
    if (updateCode != null && !updateCode.trim().isEmpty()) {
        try {
            String mcpValidation = mcpServer.validateUpdate(updateCode);
            if (mcpValidation.equals("VALID")) {
                String currentSource = getCurrentIDESource();
                String updatedSource = insertUpdateCode(currentSource, updateCode);
                
                Files.write(Paths.get("MCPIntegratedIDE.java"), updatedSource.getBytes());
                
                Process compile = Runtime.getRuntime().exec("javac MCPIntegratedIDE.java");
                compile.waitFor();
                
                if (compile.exitValue() == 0) {
                    log("MCP-validated IDE update successful!");
                    askAllAIsMCP("IDE updated with MCP validation: " + updateCode);
                } else {
                    log("Compilation failed. Update rejected.");
                }
            } else {
                log("MCP validation failed: " + mcpValidation);
            }
            
        } catch (Exception e) {
            log("Error updating IDE: " + e.getMessage());
        }
    }
}

private String insertUpdateCode(String source, String updateCode) {
    int lastBrace = source.lastIndexOf("}");
    return source.substring(0, lastBrace) + 
           "\n    // MCP-VALIDATED UPDATE\n    " + 
           updateCode + "\n" + 
           source.substring(lastBrace);
}

class MCPServer {
    private boolean running = false;
    private Map<String, String> tools = new HashMap<>();
    private Map<String, String> resources = new HashMap<>();
    
    public void start() {
        running = true;
        initializeTools();
        log("MCP Server started with tools and resources");
    }
    
    private void initializeTools() {
        tools.put("code_analyzer", "Analyzes code structure and quality");
        tools.put("file_manager", "Manages project files and directories");
        tools.put("git_helper", "Git operations and version control");
        tools.put("compiler_assistant", "Compilation and build assistance");
        tools.put("test_generator", "Generates unit tests");
    }
    
    public void connectClient(String clientUrl) {
        log("MCP Client connected: " + clientUrl);
    }
    
    public String[] listTools() {
        return tools.keySet().toArray(new String[0]);
    }
    
    public String executeTool(String toolName, String params) {
        if (tools.containsKey(toolName)) {
            return "MCP Tool '" + toolName + "' executed with params: " + params;
        }
        return "Tool not found: " + toolName;
    }
    
    public String getCodeContext(String code) {
        return "MCP Context: Java code, " + code.split("\n").length + " lines, contains main method: " + code.contains("main");
    }
    
    public String getAvailableTools() {
        return String.join(", ", tools.keySet());
    }
    
    public void notifyFileChange(String fileName, String content) {
        log("MCP: File change notification - " + fileName);
    }
    
    public String validateUpdate(String code) {
        if (code.contains("System.exit") || code.contains("Runtime.exec")) {
            return "INVALID: Potentially dangerous code";
        }
        return "VALID";
    }
}

public final class GitHubDeviceFlow {

    private static final String CLIENT_ID = "Iv1.8a61f9b3a7aba766"; // GitHub public Copilot ID
    public static CopilotAuth login(Component parent) throws IOException {
        OkHttpClient http = new OkHttpClient();

        /* 1. request device code */
        RequestBody body = new FormBody.Builder()
                .add("client_id", CLIENT_ID)
                .add("scope", "copilot")
                .build();
        Request req = new Request.Builder()
                .url("https://github.com/login/device/code")
                .post(body)
                .header("Accept", "application/json")
                .build();
        try (Response rsp = http.newCall(req).execute()) {
            DeviceCode dc = Jackson.readValue(rsp.body().byteStream(), DeviceCode.class);

            /* 2. show user the code */
            String msg = """
                         1. Browse  :  %s
                         2. Enter code :  %s
                         3. Press Authorize
                         4. Wait… (this window will auto-close)
                         """.formatted(dc.verification_uri, dc.user_code);
            JLabel label = new JLabel("<html><pre>" + msg);
            JDialog dlg = new JOptionPane(label, JOptionPane.INFORMATION_MESSAGE)
                    .createDialog(parent, "GitHub Copilot OAuth");
            dlg.setModal(false);
            dlg.setVisible(true);

            /* 3. poll for token */
            CopilotAuth auth = pollToken(http, dc, dlg);
            dlg.dispose();
            return auth;
        }
    }

    private static CopilotAuth pollToken(OkHttpClient http, DeviceCode dc, JDialog dlg) throws IOException {
        Instant dead = Instant.now().plusSeconds(dc.expires_in);
        while (Instant.now().isBefore(dead)) {
            RequestBody body = new FormBody.Builder()
                    .add("client_id", CLIENT_ID)
                    .add("device_code", dc.device_code)
                    .add("grant_type", "urn:ietf:params:oauth:grant-type:device_code")
                    .build();
            Request req = new Request.Builder()
                    .url("https://github.com/login/oauth/access_token")
                    .post(body)
                    .header("Accept", "application/json")
                    .build();
            try (Response rsp = http.newCall(req).execute()) {
                TokenResponse tr = Jackson.readValue(rsp.body().byteStream(), TokenResponse.class);
                if (tr.access_token != null) {
                    return new CopilotAuth(
                            "https://api.github.com/copilot",
                            "bearer",
                            tr.access_token,
                            tr.refresh_token,
                            Instant.now().plusSeconds(tr.expires_in).getEpochSecond()
                    );
                }
                if ("authorization_pending".equals(tr.error)) {
                    sleep(dc.interval);
                    continue;
                }
                throw new IOException(tr.error + ": " + tr.error_description);
            }
        }
        throw new IOException("device code expired");
    }

    /* Jackson DTOs */
    private static class DeviceCode {
        public String device_code, user_code, verification_uri;
        public int expires_in, interval;
    }
    private static class TokenResponse {
        public String access_token, refresh_token, error, error_description;
        public int expires_in;
    }
    private static void sleep(int s) { try { Thread.sleep(s*1000L); } catch (InterruptedException ignore) {} }
}

public final class CopilotService {

    private static CopilotService INSTANCE;
    public  static CopilotService get() { return INSTANCE; }

    private final OkHttpClient http = new OkHttpClient.Builder()
            .connectTimeout(10, TimeUnit.SECONDS)
            .build();
    private final Preference prefs = Preferences.userNodeForPackage(HybridIDE.class);
    private CopilotAuth auth;

    public CopilotService() {
        INSTANCE = this;
        auth = loadAuth();
    }

    /* ---------- public API ---------- */

    public boolean isSignedIn() { return auth != null && Instant.now().getEpochSecond() < auth.expiresAtEpochS(); }

    public void signIn(Component parent) throws IOException {
        auth = GitHubDeviceFlow.login(parent);
        storeAuth(auth);
    }

    public void signOut() {
        auth = null;
        prefs.remove("copilot");
    }

    /** Stream completion into the supplied consumer (UI thread) */
    public void complete(String languageId, String prompt, Consumer<String> onDelta, Runnable onDone) {
        if (!isSignedIn()) { SwingUtilities.invokeLater(() -> onDelta.accept("/* 🔒 not signed in */")); return; }

        String json = """
                      {
                        "model": "copilot-codex",
                        "prompt": "%s",
                        "max_tokens": 128,
                        "temperature": 0.2,
                        "stream": true
                      }
                      """.formatted(Json.escape(prompt));

        Request req = new Request.Builder()
                .url(auth.host() + "/completions")
                .post(RequestBody.create(json, MediaType.parse("application/json")))
                .header("Authorization", auth.tokenType() + " " + auth.accessToken())
                .header("Accept", "text/event-stream")
                .build();

        http.newCall(req).enqueue(new Callback() {
            @Override public void onFailure(Call call, IOException e) {
                SwingUtilities.invokeLater(() -> onDelta.accept("/* ⚠ " + e.getMessage() + " */"));
            }
            @Override public void onResponse(Call call, Response rsp) throws IOException {
                try (ResponseBody body = rsp.body()) {
                    BufferedSource src = body.source();
                    String line;
                    while ((line = src.readUtf8Line()) != null) {
                        if (line.startsWith("data: ")) {
                            String data = line.substring(6);
                            if ("[DONE]".equals(data)) { SwingUtilities.invokeLater(onDone); return; }
                            Delta d = Jackson.readValue(data, Delta.class);
                            SwingUtilities.invokeLater(() -> onDelta.accept(d.choices[0].text));
                        }
                    }
                }
            }
        });
    }

    /* ---------- helpers ---------- */
    private CopilotAuth loadAuth() {
        try {
            byte[] bytes = prefs.getByteArray("copilot", null);
            return bytes == null ? null : Jackson.readValue(bytes, CopilotAuth.class);
        } catch (Exception ignore) { return null; }
    }
    private void storeAuth(CopilotAuth a) throws IOException {
        prefs.putByteArray("copilot", Jackson.writeValueAsBytes(a));
    }

    /* Jackson DTO */
    private static class Delta {
        public Choice[] choices;
        static class Choice { public String text; }
    }
}

import { editor } from 'monaco-editor';

export function installCopilot(ed: editor.IStandaloneCodeEditor) {
    let controller = new AbortController();

    ed.onDidChangeModelContent(e => {
        controller.abort();          // cancel previous stream
        controller = new AbortController();

        const pos = ed.getPosition()!;
        const prompt = ed.getModel()!.getValueInRange({
            startLineNumber: Math.max(1, pos.lineNumber - 20),
            startColumn: 1,
            endLineNumber: pos.lineNumber,
            endColumn: pos.column
        });

        // send to Swing backend
        (window as any).bridge.send('copilot', { language: 'java', prompt });
    });

    // listen for deltas coming back
    (window as any).bridge.on('copilot.delta', (payload: { text: string }) => {
        const insert = { range: ed.getSelection()!, text: payload.text };
        ed.executeEdits('copilot', [insert]);
    });
}
