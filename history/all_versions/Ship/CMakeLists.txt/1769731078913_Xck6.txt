# CMakeLists.txt - RawrXD Agent Build Configuration
# Pure C++20 / Win32 - Zero Qt Dependencies

cmake_minimum_required(VERSION 3.20)
project(RawrXD_Agent VERSION 1.0.0 LANGUAGES CXX)

# C++20 standard
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Windows-specific settings
if(WIN32)
    add_definitions(-DUNICODE -D_UNICODE -DWIN32_LEAN_AND_MEAN -DNOMINMAX)
    set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")
endif()

# Source files
set(AGENT_HEADERS
    agent_kernel_main.hpp
    QtReplacements.hpp
    ToolExecutionEngine.hpp
    LLMClient.hpp
    AgentOrchestrator.hpp
    ToolImplementations.hpp
    Win32UIIntegration.hpp
    RawrXD_Agent.hpp
)

# Console Agent
add_executable(RawrXD_Agent_Console
    Integration.cpp
    ${AGENT_HEADERS}
)

target_compile_definitions(RawrXD_Agent_Console PRIVATE
    _CONSOLE
)

target_link_libraries(RawrXD_Agent_Console PRIVATE
    winhttp
    ws2_32
    comctl32
    ole32
    shell32
    shlwapi
)

# GUI Agent
add_executable(RawrXD_Agent_GUI WIN32
    Integration.cpp
    ${AGENT_HEADERS}
)

target_link_libraries(RawrXD_Agent_GUI PRIVATE
    winhttp
    ws2_32
    comctl32
    ole32
    shell32
    shlwapi
)

# Example Application
add_executable(RawrXD_Agent_Example
    Example_Application.cpp
    ${AGENT_HEADERS}
)

target_link_libraries(RawrXD_Agent_Example PRIVATE
    winhttp
    ws2_32
    comctl32
    ole32
    shell32
    shlwapi
)

# Test Suite
add_executable(RawrXD_Agent_Tests
    test_suite.cpp
    ${AGENT_HEADERS}
)

target_link_libraries(RawrXD_Agent_Tests PRIVATE
    winhttp
    ws2_32
    comctl32
    ole32
    shell32
    shlwapi
)

# Benchmark Suite
add_executable(RawrXD_Agent_Benchmark
    benchmark_suite.cpp
    ${AGENT_HEADERS}
)

target_link_libraries(RawrXD_Agent_Benchmark PRIVATE
    winhttp
    ws2_32
    comctl32
    ole32
    shell32
    shlwapi
)

# Install targets
install(TARGETS
    RawrXD_Agent_Console
    RawrXD_Agent_GUI
    RawrXD_Agent_Example
    RUNTIME DESTINATION bin
)

install(FILES ${AGENT_HEADERS}
    DESTINATION include/RawrXD
)

# CPack configuration
set(CPACK_PACKAGE_NAME "RawrXD_Agent")
set(CPACK_PACKAGE_VERSION ${PROJECT_VERSION})
set(CPACK_PACKAGE_VENDOR "RawrXD")
set(CPACK_PACKAGE_DESCRIPTION_SUMMARY "Autonomous AI Coding Agent")
include(CPack)
