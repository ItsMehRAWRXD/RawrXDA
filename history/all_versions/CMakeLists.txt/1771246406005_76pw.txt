cmake_minimum_required(VERSION 3.20)
project(RawrXD LANGUAGES CXX C ASM_MASM)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Enable MASM for native implementations
enable_language(ASM_MASM)
set(CMAKE_ASM_MASM_FLAGS "/nologo /W3 /Cx /Zi")

# Force x64 architecture - RawrXD is x64-ONLY
if(MSVC)
    set(CMAKE_GENERATOR_PLATFORM x64)
    set(CMAKE_SIZEOF_VOID_P 8)
    if(NOT CMAKE_GENERATOR_PLATFORM STREQUAL "x64")
        message(FATAL_ERROR "CRITICAL: CMAKE_GENERATOR_PLATFORM must be x64")
    endif()
    if(NOT CMAKE_SIZEOF_VOID_P EQUAL 8)
        message(FATAL_ERROR "CRITICAL: CMAKE_SIZEOF_VOID_P must be 8 (x64)")
    endif()
    message(STATUS "✓ 64-bit configuration validated")
endif()

# Pin MSVC runtime for deterministic builds
if(MSVC)
    set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>DLL")
    add_compile_options(
        $<$<CONFIG:Debug>:/MDd>
        $<$<CONFIG:Release>:/MD>
        $<$<CONFIG:RelWithDebInfo>:/MD>
        $<$<CONFIG:MinSizeRel>:/MD>
    )
endif()

add_definitions(-DNOMINMAX -DWIN32_LEAN_AND_MEAN -D_CRT_SECURE_NO_WARNINGS)
add_definitions(-DHAVE_D3D12=1 -DHAVE_NATIVE_QUANT=1)

# Native Windows sources only - no external dependencies
set(COMMON_SOURCES
    src/cpu_inference_engine.cpp
    src/rawrxd_inference.cpp
    src/rawrxd_tokenizer.cpp
    src/rawrxd_transformer.cpp
    src/rawrxd_sampler.cpp
    src/rawrxd_model_loader.cpp
    src/rawrxd_kernels.cpp
    src/token_generator.cpp
    src/native_quant.cpp
    src/d3d12_compute.cpp
)

set(MAIN_SOURCES
    src/main.cpp
    src/utils/settings_manager.cpp
    src/ide_orchestrator.cpp
    src/agentic_ide.cpp
    src/agentic_executor.cpp
    src/agentic_memory_system.cpp
    src/agentic_observability.cpp
    src/universal_model_router.cpp
    src/agentic_engine.cpp
    src/action_executor.cpp
    src/cloud_api_client.cpp
    src/RawrXD_PipeClient.cpp
    src/RawrXD_Renderer_D2D.cpp
    src/utils/Diagnostics.cpp
    src/utils/ErrorReporter.cpp
    src/utils/InferenceSettingsManager.cpp
    ${COMMON_SOURCES}
)

# Main Executable
add_executable(RawrXD ${MAIN_SOURCES})
target_include_directories(RawrXD PRIVATE src include)

# Self Audit Executable
add_executable(RawrXD_SelfAudit 
    src/test_self_audit.cpp 
    src/ide_auditor.cpp
    src/swarm_orchestrator.cpp 
    src/chain_of_thought.cpp 
    src/net/net_impl_win32.cpp
    ${COMMON_SOURCES}
)
target_include_directories(RawrXD_SelfAudit PRIVATE src include)

# Linking - Native Windows APIs only
if(MSVC)
    target_link_libraries(RawrXD PRIVATE ws2_32 d2d1 dwrite windowscodecs d3d12 dxgi d3dcompiler)
    target_link_libraries(RawrXD_SelfAudit PRIVATE ws2_32)
    target_compile_options(RawrXD PRIVATE /arch:AVX512 /O2 /std:c++20 /EHsc /bigobj)
    target_link_options(RawrXD PRIVATE /LARGEADDRESSAWARE)
else()
    target_link_libraries(RawrXD PRIVATE ws2_32 d2d1 dwrite windowscodecs uuid ole32 shlwapi version d3d12 dxgi d3dcompiler)
    target_link_libraries(RawrXD_SelfAudit PRIVATE ws2_32 uuid ole32 shlwapi version)
    target_compile_options(RawrXD PRIVATE -O2 -mavx512f -Wa,-mbig-obj)
    target_compile_options(RawrXD_SelfAudit PRIVATE -O2 -mavx512f -Wa,-mbig-obj)
endif()

message(STATUS "✓ CMake configuration complete - no external dependencies")

set_target_properties(RawrXD_SelfAudit PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
