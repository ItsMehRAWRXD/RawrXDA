# Performance optimization levels
set(MASM_OPTIMIZATION_LEVEL "MAXIMUM" CACHE STRING "MASM optimization level")

    "NONE" "STANDARD" "AGGRESSIVE" "MAXIMUM")

# Target architecture selection
set(MASM_TARGET_ARCH "X64" CACHE STRING "Target architecture for MASM compilation")
set_property(CACHE MASM_TARGET_ARCH PROPERTY STRINGS
    "X64" "X86" "ARM64")

# Build configuration options
option(MASM_ENABLE_AVX512 "Enable AVX-512 instruction set optimizations" ON)
option(MASM_ENABLE_LTO "Enable Link-Time Optimization for MASM objects" ON)
option(MASM_ENABLE_PGO "Enable Profile-Guided Optimization" OFF)
option(MASM_ENABLE_SANITIZERS "Enable address/memory sanitizers" OFF)
option(MASM_BUILD_TESTS "Build comprehensive test suite" ON)

# ==============================================================================
# COMPILER DETECTION & CONFIGURATION
# ==============================================================================

# Detect and configure MASM compiler
if(MSVC)
    # Find ML64.exe (MASM assembler)
    find_program(MASM_COMPILER
        NAMES ml64.exe
        PATHS
            "C:/Program Files/Microsoft Visual Studio/2022/Professional/VC/Tools/MSVC/*/bin/Hostx64/x64"
            "C:/Program Files/Microsoft Visual Studio/2022/Enterprise/VC/Tools/MSVC/*/bin/Hostx64/x64"
            "C:/Program Files/Microsoft Visual Studio/2022/Community/VC/Tools/MSVC/*/bin/Hostx64/x64"
            "C:/Program Files (x86)/Microsoft Visual Studio/2022/BuildTools/VC/Tools/MSVC/*/bin/Hostx64/x64"
        NO_DEFAULT_PATH
    )

    if(NOT MASM_COMPILER)
        message(FATAL_ERROR "ML64.exe (MASM assembler) not found. Please install Visual Studio with MASM support.")
    endif()

    # Configure MASM compiler
    set(CMAKE_ASM_MASM_COMPILER "${MASM_COMPILER}")
    set(CMAKE_ASM_MASM_COMPILER_ID "MASM")

    # Advanced MASM flags based on optimization level
    if(MASM_OPTIMIZATION_LEVEL STREQUAL "MAXIMUM")
        set(MASM_FLAGS "/nologo /Zi /c /W3 /Ta")
        if(MASM_ENABLE_AVX512)
            list(APPEND MASM_FLAGS "/arch:AVX512")
        endif()
    elseif(MASM_OPTIMIZATION_LEVEL STREQUAL "AGGRESSIVE")
        set(MASM_FLAGS "/nologo /Zi /c /W3 /Ta")
    elseif(MASM_OPTIMIZATION_LEVEL STREQUAL "STANDARD")
        set(MASM_FLAGS "/nologo /Zi /c /W2 /Ta")
    else()
        set(MASM_FLAGS "/nologo /Zi /c /Ta")
    endif()

    # Architecture-specific flags
    if(MASM_TARGET_ARCH STREQUAL "X64")
        list(APPEND MASM_FLAGS "/D_WIN64")
    elseif(MASM_TARGET_ARCH STREQUAL "ARM64")
        list(APPEND MASM_FLAGS "/D_ARM64")
    endif()

    set(CMAKE_ASM_MASM_FLAGS "${MASM_FLAGS}" CACHE STRING "MASM compiler flags")

else()
    message(FATAL_ERROR "MASM compilation requires MSVC toolchain with ML64.exe")
endif()

# ==============================================================================
# SOURCE FILE DISCOVERY & ORGANIZATION
# ==============================================================================

# Comprehensive MASM source file discovery
file(GLOB_RECURSE MASM_ALL_SOURCES
    RELATIVE ${CMAKE_CURRENT_SOURCE_DIR}
    "*.asm"
    "*.inc"
)

# Categorize sources by functionality
set(MASM_CORE_SOURCES)
set(MASM_TENSOR_SOURCES)
set(MASM_NETWORK_SOURCES)
set(MASM_CRYPTO_SOURCES)
set(MASM_UI_SOURCES)
set(MASM_TEST_SOURCES)

foreach(source ${MASM_ALL_SOURCES})
    if(source MATCHES ".*_test\\.asm$|.*test_.*\\.asm$")
        list(APPEND MASM_TEST_SOURCES ${source})
    elseif(source MATCHES ".*tensor.*\\.asm$|.*neural.*\\.asm$|.*ml.*\\.asm$")
        list(APPEND MASM_TENSOR_SOURCES ${source})
    elseif(source MATCHES ".*network.*\\.asm$|.*http.*\\.asm$|.*socket.*\\.asm$")
        list(APPEND MASM_NETWORK_SOURCES ${source})
    elseif(source MATCHES ".*crypto.*\\.asm$|.*encrypt.*\\.asm$|.*hash.*\\.asm$")
        list(APPEND MASM_CRYPTO_SOURCES ${source})
    elseif(source MATCHES ".*ui.*\\.asm$|.*gui.*\\.asm$|.*render.*\\.asm$")
        list(APPEND MASM_UI_SOURCES ${source})
    else()
        list(APPEND MASM_CORE_SOURCES ${source})
    endif()
endforeach()

    endif()
endforeach()

# ==============================================================================
# LIBRARY TARGETS - ENTERPRISE GRADE
# ==============================================================================

# Core MASM library with maximum optimization
add_library(masm_core STATIC ${MASM_CORE_SOURCES})
target_compile_options(masm_core PRIVATE
    $<$<COMPILE_LANGUAGE:ASM_MASM>:/favor:AMD64>
    $<$<COMPILE_LANGUAGE:ASM_MASM>:/Qpar>  # Auto-parallelization
)

# Tensor/ML optimized library
add_library(masm_tensor STATIC ${MASM_TENSOR_SOURCES})
if(MASM_ENABLE_AVX512)
    target_compile_options(masm_tensor PRIVATE
        $<$<COMPILE_LANGUAGE:ASM_MASM>:/arch:AVX512>
    )
endif()

# Network library with security optimizations
add_library(masm_network STATIC ${MASM_NETWORK_SOURCES})
target_compile_options(masm_network PRIVATE
    $<$<COMPILE_LANGUAGE:ASM_MASM>:/GS>   # Buffer security checks
    $<$<COMPILE_LANGUAGE:ASM_MASM>:/sdl>  # Additional security checks
)

# Cryptography library
add_library(masm_crypto STATIC ${MASM_CRYPTO_SOURCES})
target_compile_options(masm_crypto PRIVATE
    $<$<COMPILE_LANGUAGE:ASM_MASM>:/guard:cf>  # Control Flow Guard
)

# UI/GUI library
add_library(masm_ui STATIC ${MASM_UI_SOURCES})

# ==============================================================================
# MAIN MASM COMPONENTS LIBRARY
# ==============================================================================

# Set language property for all MASM sources
set_source_files_properties(${MASM_PRODUCTION_SOURCES} PROPERTIES LANGUAGE ASM_MASM)

# Create the main MASM components library
add_library(masm_components STATIC ${MASM_PRODUCTION_SOURCES})

# Advanced target properties
target_compile_definitions(masm_components PRIVATE
    MASM_VERSION_MAJOR=3
    MASM_VERSION_MINOR=0
    MASM_TARGET_ARCH_${MASM_TARGET_ARCH}
    $<$<BOOL:${MASM_ENABLE_AVX512}>:MASM_AVX512_ENABLED>
)

# Include directories for MASM compilation
target_include_directories(masm_components PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/../include
    ${CMAKE_CURRENT_SOURCE_DIR}/interconnect
)

# Link dependencies
target_link_libraries(masm_components PUBLIC
    masm_core
    masm_tensor
    masm_network
    $<$<BOOL:${MASM_ENABLE_AVX512}:masm_crypto>
)

# ==============================================================================
# OPTIMIZATION & PERFORMANCE CONFIGURATION
# ==============================================================================

# Link-Time Optimization (LTO)
if(MASM_ENABLE_LTO AND MSVC)
    set_target_properties(masm_components PROPERTIES
        INTERPROCEDURAL_OPTIMIZATION TRUE
    )
endif()

# Profile-Guided Optimization (PGO)
if(MASM_ENABLE_PGO)
    # Instrumentation phase
    add_library(masm_components_pgo_instrument STATIC ${MASM_PRODUCTION_SOURCES})
    set_target_properties(masm_components_pgo_instrument PROPERTIES
        COMPILE_FLAGS "/GL /LTCG:PGINSTRUMENT"
        LINK_FLAGS "/LTCG:PGINSTRUMENT"
    )

    # Optimized phase (requires .pgd file from instrumentation run)
    set_target_properties(masm_components PROPERTIES
        COMPILE_FLAGS "/GL /LTCG:PGOPTIMIZE"
        LINK_FLAGS "/LTCG:PGOPTIMIZE"
    )
endif()

# Sanitizer integration
if(MASM_ENABLE_SANITIZERS)
    target_compile_options(masm_components PRIVATE
        $<$<COMPILE_LANGUAGE:ASM_MASM>:/fsanitize=address>
    )
    target_link_options(masm_components PRIVATE
        /fsanitize=address
    )
endif()

# ==============================================================================
# INSTALLATION & DEPLOYMENT
# ==============================================================================

if(MASM_ENABLE_INSTALL)
    # Install libraries
    install(TARGETS masm_components masm_core masm_tensor masm_network
        ARCHIVE DESTINATION lib
        LIBRARY DESTINATION lib
        RUNTIME DESTINATION bin
    )

    # Install headers and include files
    install(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/
        DESTINATION include/masm
        FILES_MATCHING PATTERN "*.inc"
    )

    # Install documentation
    if(MASM_ENABLE_DOCS AND Doxygen_FOUND)
        install(DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/docs/
            DESTINATION share/doc/masm
        )
    endif()

    # Install CMake config files
    include(CMakePackageConfigHelpers)
    write_basic_package_version_file(
        "${CMAKE_CURRENT_BINARY_DIR}/MASMConfigVersion.cmake"
        VERSION 3.0.0
        COMPATIBILITY SameMajorVersion
    )

    configure_package_config_file(
        "${CMAKE_CURRENT_SOURCE_DIR}/MASMConfig.cmake.in"
        "${CMAKE_CURRENT_BINARY_DIR}/MASMConfig.cmake"
        INSTALL_DESTINATION lib/cmake/MASM
    )

    install(FILES
        "${CMAKE_CURRENT_BINARY_DIR}/MASMConfig.cmake"
        "${CMAKE_CURRENT_BINARY_DIR}/MASMConfigVersion.cmake"
        DESTINATION lib/cmake/MASM
    )
endif()
