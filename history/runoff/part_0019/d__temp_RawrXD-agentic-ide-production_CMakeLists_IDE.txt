cmake_minimum_required(VERSION 3.20)
project(RawrXD-IDE VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Qt6 configuration
find_package(Qt6 COMPONENTS Core Gui Widgets Concurrent REQUIRED)

# Source files
set(SOURCES
    src/ModelLoaderBridge.cpp
    src/ModelCacheManager.cpp
    src/InferenceSession.cpp
    src/TokenStreamRouter.cpp
    src/ModelSelectionDialog.cpp
    src/IDEIntegration.cpp
    src/PerformanceMonitor.cpp
    src/ModelMetadataParser.cpp
    src/StreamingInferenceEngine.cpp
)

set(HEADERS
    src/ModelLoaderBridge.h
    src/ModelCacheManager.h
    src/InferenceSession.h
    src/TokenStreamRouter.h
    src/ModelSelectionDialog.h
    src/IDEIntegration.h
    src/PerformanceMonitor.h
    src/ModelMetadataParser.h
    src/StreamingInferenceEngine.h
)

# Create executable
add_executable(RawrXD-IDE 
    ${SOURCES}
    ${HEADERS}
)

# Include directories
target_include_directories(RawrXD-IDE PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/src
    ${CMAKE_CURRENT_SOURCE_DIR}/RawrXD-ModelLoader/include
    ${CMAKE_CURRENT_SOURCE_DIR}/RawrXD-ModelLoader/src
)

# Qt configuration
target_link_libraries(RawrXD-IDE
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
    Qt6::Concurrent
)

# Sovereign Loader DLL linking
link_directories(${CMAKE_CURRENT_SOURCE_DIR}/build-sovereign-static/bin)
target_link_libraries(RawrXD-IDE 
    RawrXD-SovereignLoader
)

# Compiler flags for performance
if(MSVC)
    target_compile_options(RawrXD-IDE PRIVATE
        /arch:AVX512
        /O2
        /W4
        /Oi
        /Ot
        /GL
        /fp:fast
    )
    target_link_options(RawrXD-IDE PRIVATE /LTCG)
else()
    target_compile_options(RawrXD-IDE PRIVATE
        -mavx512f
        -O3
        -Wall
        -ffast-math
    )
endif()

# Version macros
target_compile_definitions(RawrXD-IDE PRIVATE
    QT_DEPRECATED_WARNINGS
    QT_NO_WARNING_OUTPUT
    RAWRXD_VERSION="1.0.0"
)

# Post-build: Copy DLL to output
add_custom_command(TARGET RawrXD-IDE POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        ${CMAKE_CURRENT_SOURCE_DIR}/build-sovereign-static/bin/RawrXD-SovereignLoader.dll
        $<TARGET_FILE_DIR:RawrXD-IDE>/RawrXD-SovereignLoader.dll
    COMMENT "Copying Sovereign Loader DLL to output directory"
)

# Installation
install(TARGETS RawrXD-IDE DESTINATION bin)

message(STATUS "========================================")
message(STATUS "RawrXD-IDE v1.0 (Production Release)")
message(STATUS "========================================")
message(STATUS "C++ Standard: ${CMAKE_CXX_STANDARD}")
message(STATUS "Compiler: ${CMAKE_CXX_COMPILER}")
message(STATUS "Qt Version: ${Qt6_VERSION}")
message(STATUS "Build Type: ${CMAKE_BUILD_TYPE}")
message(STATUS "========================================")
