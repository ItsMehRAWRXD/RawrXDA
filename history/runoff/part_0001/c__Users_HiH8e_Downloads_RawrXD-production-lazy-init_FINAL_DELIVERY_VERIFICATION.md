## 💻 QUICK CODE EXAMPLE

### Hello World with Qt String Wrapper
```cpp
#include "qt_string_wrapper_masm.hpp"

int main() {
    // Create wrapper
    QtStringWrapperMASM wrapper;
    
    // Create a string
    void* qstring = wrapper_qstring_create();
    
    // Append text
    wrapper.appendToString(qstring, "Hello");
    wrapper.appendToString(qstring, " World!");
    
    // Get as QString
    QString result = wrapper.getStringAsQString(qstring);
    qDebug() << result;  // Output: "Hello World!"
    
    // Clean up
    wrapper.deleteString(qstring);
    
    return 0;
}
```
