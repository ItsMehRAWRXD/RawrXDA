cmake_minimum_required(VERSION 3.16)
project(test_chat_integration)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_AUTOMOC ON)

# Find Qt6
find_package(Qt6 COMPONENTS Core Gui Widgets REQUIRED)

# Include directories
include_directories(${CMAKE_SOURCE_DIR})
include_directories(${CMAKE_SOURCE_DIR}/src)
include_directories(${CMAKE_SOURCE_DIR}/src/qtapp)
include_directories(${CMAKE_SOURCE_DIR}/include)

# Add the test executable
add_executable(test_chat_integration
    test_chat_integration.cpp
    src/qtapp/inference_engine.cpp
    src/qtapp/gguf_loader.cpp
    src/qtapp/vocabulary_loader.cpp
)

# Link libraries
target_link_libraries(test_chat_integration PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
)
