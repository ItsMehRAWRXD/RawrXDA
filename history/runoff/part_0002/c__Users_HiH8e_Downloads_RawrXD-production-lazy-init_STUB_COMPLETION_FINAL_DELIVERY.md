## 🎓 How It Works

### Animation Example
```asm
; Start animation (300ms)
mov ecx, 300
lea rdx, [MyCallback]
call StartAnimationTimer
; eax = timer_id

; Update in loop
mov ecx, timer_id
mov edx, 33
call UpdateAnimation
; eax = progress (0-100)
```

### UI Example
```asm
; Create mode selector
mov rcx, parent_hwnd
call ui_create_mode_combo
; rax = combobox handle

; Open file dialog
lea rcx, ["*.gguf|GGUF Models"]
call ui_open_file_dialog
; rax = file path or 0
```

### Feature Example
```asm
; Load configuration
lea rcx, ["feature_configuration.json"]
call LoadUserFeatureConfiguration

; Apply initial state
call ApplyInitialFeatureConfiguration

; Change feature
mov ecx, 5           ; feature ID
mov edx, 1           ; enable
call HandleFeatureStateChange
```

### Model Example
```asm
; Load GGUF
lea rcx, ["model.gguf"]
call rawr1024_direct_load
; rax = model pointer

; Quantize
mov ecx, [rax]       ; model
mov edx, 4           ; 4-bit
call rawr1024_quantize_model

; Get tensor
lea rcx, ["attention.0.q_proj.weight"]
call ml_masm_get_tensor
; rax = tensor pointer
```
