# RawrXD Core Library
# Shared functionality between GUI and CLI executables

set(CORE_SOURCES
    orchestra_manager.cpp
    safe_mode_config.cpp
    ../lazy_directory_loader.cpp
)

set(CORE_HEADERS
    orchestra_manager.hpp
    safe_mode_config.hpp
    ../lazy_directory_loader.h
)

# Create the shared core library (static)
add_library(RawrXD-Core STATIC ${CORE_SOURCES} ${CORE_HEADERS})

# Find required packages
find_package(Qt6 REQUIRED COMPONENTS Core Concurrent)

# Link Qt Core
target_link_libraries(RawrXD-Core PUBLIC
    Qt6::Core
    Qt6::Concurrent
)

# Include directories
target_include_directories(RawrXD-Core PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}
)

# Enable Qt MOC
set_target_properties(RawrXD-Core PROPERTIES
    AUTOMOC ON
)

# Additional platform-specific libraries
if(WIN32)
    target_link_libraries(RawrXD-Core PRIVATE
        ws2_32
        advapi32
    )
endif()

# zlib integration (if available)
find_package(ZLIB)
if(ZLIB_FOUND)
    target_link_libraries(RawrXD-Core PRIVATE ZLIB::ZLIB)
    target_compile_definitions(RawrXD-Core PRIVATE HAVE_ZLIB=1)

# ════════════════════════════════════════════════════════════════════════════════
# RawrXD LazyInit Main System Spine (Pure MASM64)
# ════════════════════════════════════════════════════════════════════════════════

# Enable MASM64 if not already enabled
enable_language(ASM_MASM)

# Main executable (console application)
add_executable(rawrxd_lazyinit_main
    RawrXD_LazyInit_Main.asm
)

set_property(SOURCE RawrXD_LazyInit_Main.asm PROPERTY LANGUAGE ASM_MASM)

# Pure MASM64 executable configuration
set_target_properties(rawrxd_lazyinit_main PROPERTIES
    WIN32_EXECUTABLE FALSE
    LINK_FLAGS "/ENTRY:main /SUBSYSTEM:CONSOLE /NODEFAULTLIB /STACK:0x100000,0x100000"
)

target_link_libraries(rawrxd_lazyinit_main PRIVATE kernel32 user32)

# DLL version (for hot-injection)
add_library(rawrxd_lazyinit_dll SHARED
    RawrXD_LazyInit_Main.asm
)

set_property(SOURCE RawrXD_LazyInit_Main.asm PROPERTY LANGUAGE ASM_MASM)

set_target_properties(rawrxd_lazyinit_dll PROPERTIES
    OUTPUT_NAME "RawrXD_LazyInit_Main"
    PREFIX ""
    LINK_FLAGS "/DLL /ENTRY:DllMain /NODEFAULTLIB /STACK:0x100000,0x100000"
)

target_link_libraries(rawrxd_lazyinit_dll PRIVATE kernel32 user32)

# Install LazyInit targets
install(TARGETS rawrxd_lazyinit_main rawrxd_lazyinit_dll
    RUNTIME DESTINATION bin
    LIBRARY DESTINATION lib
)
else()
    message(STATUS "zlib not found, using stub implementations")
endif()
