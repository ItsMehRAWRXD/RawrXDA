cmake_minimum_required(VERSION 3.20)

# Qt UI scaffold
set(QTAPP_SOURCES
    ${QTAPP_SOURCES}
    ${CMAKE_CURRENT_SOURCE_DIR}/MainWindow.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/MainWindow.h
)

find_package(Qt6 REQUIRED COMPONENTS Widgets)

add_library(qtapp_ui OBJECT ${QTAPP_SOURCES})
target_link_libraries(qtapp_ui PRIVATE Qt6::Widgets)
target_compile_features(qtapp_ui PRIVATE cxx_std_20)
