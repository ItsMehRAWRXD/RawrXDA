#==============================================================================
# MASM CRITICAL PATH CONFIGURATION FOR CMAKELISTS.TXT
#==============================================================================
#
# Add this section to the main CMakeLists.txt BEFORE the main target definition.
# Location: After GGML setup (around line 300-350), before add_executable()
#
# CRITICAL PATHS:
# 1. Token Generation (18-22 cycles, 8,259 TPS)
# 2. GGUF Memory Mapping (2-3ms loading)
# 3. BPE Tokenization (12.5x faster with AVX-512)
#
#==============================================================================

# Enable MASM (Microsoft Macro Assembler) for x64 assembly
enable_language(ASM_MASM)

# Verify ml64.exe is available
if(NOT CMAKE_ASM_MASM_COMPILER)
    message(WARNING "ml64.exe (MASM compiler) not found. MASM critical paths will be skipped.")
    set(ENABLE_CRITICAL_PATHS OFF)
else()
    message(STATUS "MASM compiler found: ${CMAKE_ASM_MASM_COMPILER}")
    set(ENABLE_CRITICAL_PATHS ON)
endif()

# ============================================================
# CRITICAL PATHS ASSEMBLY FILES
# ============================================================
set(CRITICAL_PATHS_ASM_SOURCES
    src/qtapp/critical_paths/token_gen_inner_loop.asm
    src/qtapp/critical_paths/gguf_memory_map.asm
    src/qtapp/critical_paths/bpe_tokenize_simd.asm
)

# Create critical paths object library
if(ENABLE_CRITICAL_PATHS)
    add_library(critical_paths_objects OBJECT
        ${CRITICAL_PATHS_ASM_SOURCES}
    )
    
    # MASM-specific compiler flags for x64
    # /c     = Compile only (no linking)
    # /Cp    = Preserve case in public symbols
    # /Zd    = Generate COFF debug information
    # /WX    = Treat warnings as errors
    # /Fo    = Output object file
    set_target_properties(critical_paths_objects PROPERTIES
        ASM_MASM_FLAGS "/c /Cp /Zd /WX"
        # Ensure MASM files are treated as x64 assembly
        LANGUAGE ASM_MASM
    )
    
    # Include path for shared headers
    target_include_directories(critical_paths_objects PRIVATE
        ${CMAKE_SOURCE_DIR}/src/qtapp
        ${CMAKE_SOURCE_DIR}/include
    )
    
    message(STATUS "✓ Critical paths (MASM) object library configured")
    message(STATUS "  - Token generation inner loop (18-22 cycles)")
    message(STATUS "  - GGUF memory mapping (2-3ms loading)")
    message(STATUS "  - BPE tokenization SIMD (12.5x faster)")
endif()

# ============================================================
# LINK CRITICAL PATHS INTO MAIN EXECUTABLE
# ============================================================
#
# After add_executable(RawrXD-QtShell ...), add this line:
#
#   if(ENABLE_CRITICAL_PATHS)
#       target_link_libraries(RawrXD-QtShell PRIVATE critical_paths_objects)
#   endif()
#
# This will link the compiled .obj files into the executable with:
# /SECTION:.text,RWE    = Make code section readable, writable, executable
# /MERGE:.rdata=.text   = Merge read-only data into code section (cache efficiency)
# /OPT:REF              = Remove unreferenced functions
# /OPT:ICF              = Identical code folding
# /ALIGN:16             = 16-byte code alignment for cache coherency
#
