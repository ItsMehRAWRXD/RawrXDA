# CMakeLists_template_generator.txt
# CMake configuration for the universal template generator
# Integrates with the existing RawrZApp build system

cmake_minimum_required(VERSION 3.16)
project(TemplateGenerator)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Platform detection
if(WIN32)
    set(PLATFORM_WINDOWS TRUE)
elseif(APPLE)
    set(PLATFORM_MACOS TRUE)
elseif(UNIX)
    set(PLATFORM_LINUX TRUE)
endif()

# Architecture detection
if(CMAKE_SIZEOF_VOID_P EQUAL 8)
    set(ARCH_64BIT TRUE)
else()
    set(ARCH_32BIT TRUE)
endif()

# Compiler detection
if(CMAKE_CXX_COMPILER_ID STREQUAL "GNU")
    set(COMPILER_GCC TRUE)
elseif(CMAKE_CXX_COMPILER_ID STREQUAL "Clang")
    set(COMPILER_CLANG TRUE)
elseif(CMAKE_CXX_COMPILER_ID STREQUAL "MSVC")
    set(COMPILER_MSVC TRUE)
endif()

# Assembly source files for template generator
set(TEMPLATE_GENERATOR_ASM_SOURCES
    data_source_adapters.asm
    eon_template_lexer.asm
    eon_template_parser.asm
    eon_template_engine.asm
    uim_model.asm
    language_generators.asm
    template_generator_main.asm
)

# C++ wrapper sources (if needed)
set(TEMPLATE_GENERATOR_CPP_SOURCES
    template_generator_wrapper.cpp
)

# Include directories
include_directories(${CMAKE_CURRENT_SOURCE_DIR})
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/../)

# Compiler flags
if(COMPILER_GCC OR COMPILER_CLANG)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -O2")
    set(CMAKE_ASM_FLAGS "${CMAKE_ASM_FLAGS} -felf64")
elseif(COMPILER_MSVC)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /W3 /O2")
endif()

# Platform-specific settings
if(PLATFORM_WINDOWS)
    set(CMAKE_ASM_FLAGS "${CMAKE_ASM_FLAGS} -f win64")
    set(EXECUTABLE_EXTENSION ".exe")
elseif(PLATFORM_MACOS)
    set(CMAKE_ASM_FLAGS "${CMAKE_ASM_FLAGS} -f macho64")
    set(EXECUTABLE_EXTENSION "")
elseif(PLATFORM_LINUX)
    set(CMAKE_ASM_FLAGS "${CMAKE_ASM_FLAGS} -f elf64")
    set(EXECUTABLE_EXTENSION "")
endif()

# Create template generator executable
add_executable(template_generator
    ${TEMPLATE_GENERATOR_ASM_SOURCES}
    ${TEMPLATE_GENERATOR_CPP_SOURCES}
)

# Set output name
set_target_properties(template_generator PROPERTIES
    OUTPUT_NAME "template_generator${EXECUTABLE_EXTENSION}"
)

# Link libraries
if(PLATFORM_LINUX)
    target_link_libraries(template_generator m)
endif()

# Custom command to generate code from templates
function(add_template_generation target_name)
    set(options)
    set(oneValueArgs SOURCE TEMPLATE OUTPUT LANGUAGE)
    set(multiValueArgs DEPENDS)
    cmake_parse_arguments(TEMPLATE "${options}" "${oneValueArgs}" "${multiValueArgs}" ${ARGN})
    
    # Validate arguments
    if(NOT TEMPLATE_SOURCE)
        message(FATAL_ERROR "SOURCE is required for template generation")
    endif()
    if(NOT TEMPLATE_TEMPLATE)
        message(FATAL_ERROR "TEMPLATE is required for template generation")
    endif()
    if(NOT TEMPLATE_OUTPUT)
        message(FATAL_ERROR "OUTPUT is required for template generation")
    endif()
    if(NOT TEMPLATE_LANGUAGE)
        set(TEMPLATE_LANGUAGE "eon")
    endif()
    
    # Create custom command
    add_custom_command(
        OUTPUT ${TEMPLATE_OUTPUT}
        COMMAND template_generator
        ARGS
            --source ${TEMPLATE_SOURCE}
            --template ${TEMPLATE_TEMPLATE}
            --output ${TEMPLATE_OUTPUT}
            --language ${TEMPLATE_LANGUAGE}
        DEPENDS
            template_generator
            ${TEMPLATE_SOURCE}
            ${TEMPLATE_TEMPLATE}
            ${TEMPLATE_DEPENDS}
        COMMENT "Generating ${TEMPLATE_OUTPUT} from ${TEMPLATE_SOURCE} using ${TEMPLATE_TEMPLATE}"
        VERBATIM
    )
    
    # Add generated file to target sources
    target_sources(${target_name} PRIVATE ${TEMPLATE_OUTPUT})
endfunction()

# Example usage of template generation

    DEPENDS

    LIBRARY DESTINATION lib
