167 ; --- Stream callback stub (writes to debug output) ---
168 StreamTokenToIDE PROC FRAME
169     push rbp
170     .pushreg rbp
171     sub rsp, 16
172     .allocstack 16
173     .endprolog
174     ; RCX contains character
175     movzx eax, ecx
176     push rcx
177     mov rcx, rsp
178     mov byte ptr [rcx], al
179     mov byte ptr [rcx+1], 0
180     call OutputDebugStringA
181     pop rcx
182     add rsp, 16
183     pop rbp
184     ret
185 StreamTokenToIDE ENDP

189 BuildDisplayText PROC FRAME

192     sub rsp, 32

194     .endprolog
195     
196     ; Clear buffer

200

245 AppendString PROC
246     ; RSI = source, RDI = dest
247 @@:
248     mov al, [rsi]
249     test al, al
250     jz @f
251     mov [rdi], al
252     inc rsi
253     inc rdi
254     jmp @b
255 @@:
256     ret
257 AppendString ENDP

259 ; --- Autonomous Pipeline Routines ---
260 RunLocalModel PROC FRAME
261     push rbp
262     .pushreg rbp
263     sub rsp, 64
264     .allocstack 64
265     .endprolog
266 
267     mov rsi, rcx            ; prompt pointer
268 @@loopG:
269     mov al, [rsi]
270     cmp al,0
271     je @@doneG
272     movzx rcx, al
273     call StreamTokenToIDE
274     inc rsi
275     jmp @@loopG
276 @@doneG:
277     mov eax,1
278 
279     add rsp,64
280     pop rbp
281     ret
282 RunLocalModel ENDP

284 ExportTelemetry PROC FRAME
285     push rbp

305 RunSingleCycle PROC FRAME
306     push rbp
307     .pushreg rbp
308     sub rsp,32
309     .allocstack 32
310     .endprolog
311 
312     ; simulate stage updates
313     or qword ptr [g_StageMask], STAGE_COORDINATE
314     or qword ptr [g_StageMask], STAGE_PIPELINE
315 
316     lea rcx, szModelOutput
317     call RunLocalModel
318     call ExportTelemetry
319 
320     mov eax,1
321     add rsp,32
322     pop rbp
323     ret
324 RunSingleCycle ENDP

326 ; --- Window Procedure ---
327 WndProc PROC FRAME
328     push rbp
329     .pushreg rbp
330     sub rsp, 96
331     .allocstack 96
332     .endprolog
333     
334     ; Save parameters to local variables (RCX=hwnd, RDX=uMsg, R8=wParam, R9=lParam)
335     mov [rbp+16], rcx  ; hwnd
336     mov [rbp+24], rdx  ; uMsg
337     mov [rbp+32], r8   ; wParam
338     mov [rbp+40], r9   ; lParam
339     
340     mov eax, edx ; uMsg
341     
342     cmp eax, WM_CREATE
343     je HandleCreate
344     
345     cmp eax, WM_PAINT
346     je HandlePaint
347     
348     cmp eax, WM_TIMER
349     je HandleTimer
350     
351     cmp eax, WM_DESTROY
352     je HandleDestroy
353     
354     ; Default processing
355     mov rcx, [rbp+16] ; hwnd
356     mov rdx, [rbp+24] ; uMsg
357     mov r8, [rbp+32]  ; wParam
358     mov r9, [rbp+40]  ; lParam
359     call DefWindowProcA
360     jmp WndProcDone
361     
362 HandleCreate:
363     ; Set timer for autonomous updates (every 2 seconds)
364     mov rcx, [rbp+16] ; hwnd
365     mov edx, 1
366     mov r8, 2000
367     xor r9, r9
368     call SetTimer
369     xor rax, rax
370     jmp WndProcDone
371     
372 HandleTimer:
373     ; Increment cycle counter
374     lock inc g_CycleCounter
375     
376     ; run autonomous pipeline
377     call RunSingleCycle
378     
379     ; Force repaint
380     mov rcx, [rbp+16] ; hwnd
381     xor edx, edx
382     xor r8, r8
383     xor r9, r9
384     call InvalidateRect
385     xor rax, rax
386     jmp WndProcDone
387     
388 HandlePaint:
389     ; Begin paint
390     mov rcx, [rbp+16] ; hwnd
391     lea rdx, ps
392     call BeginPaint
393     mov rbx, rax ; HDC
394     
395     ; Get client rect
396     mov rcx, [rbp+16] ; hwnd
397     lea rdx, rect
398     call GetClientRect
399     
400     ; Build display text
401     call BuildDisplayText
402     
403     ; Draw text
404     mov rcx, rbx ; HDC
405     lea rdx, szDisplayBuffer
406     xor r8, r8
407     lea r9, rect
408     mov qword ptr [rsp+32], DT_LEFT
409     call DrawTextA
410     
411     ; End paint
412     mov rcx, [rbp+16] ; hwnd
413     lea rdx, ps
414     call EndPaint
415     
416     xor rax, rax
417     jmp WndProcDone
418     
419 HandleDestroy:
420     xor ecx, ecx
421     call PostQuitMessage
422     xor rax, rax
423     
424 WndProcDone:
425     add rsp, 96
426     pop rbp
427     ret
428 WndProc ENDP

430 ;=============================================================================
431 ; WINMAIN ENTRY POINT (GUI)
432 ;=============================================================================
433 WinMain PROC FRAME
434     push rbp
435     .pushreg rbp
436     sub rsp, 64
437     .allocstack 64
438     .endprolog
439     
440     ; Parameters: RCX=hInstance, RDX=hPrevInstance, R8=lpCmdLine, R9=nCmdShow
441     mov g_hInstance, rcx
442     
443     ; Register window class
444     mov wc.cbSize, SIZEOF WNDCLASSEXA
445     mov wc.style, CS_HREDRAW or CS_VREDRAW
446     lea rax, WndProc
447     mov wc.lpfnWndProc, rax
448     mov wc.cbClsExtra, 0
449     mov wc.cbWndExtra, 0
450     mov rax, rcx
451     mov wc.hInstance, rax
452     mov wc.hIcon, 0
453     
454     xor rcx, rcx
455     mov edx, IDC_ARROW
456     call LoadCursorA
457     mov wc.hCursor, rax
458     
459     mov wc.hbrBackground, COLOR_WINDOW + 1
460     mov wc.lpszMenuName, 0
461     lea rax, szClassName
462     mov wc.lpszClassName, rax
463     mov wc.hIconSm, 0
464     
465     lea rcx, wc
466     call RegisterClassExA
467     
468     ; Create window
469     xor rcx, rcx
470     lea rdx, szClassName
471     lea r8, szWindowTitle
472     mov r9d, WS_OVERLAPPEDWINDOW
473     mov dword ptr [rsp+32], CW_USEDEFAULT
474     mov dword ptr [rsp+40], CW_USEDEFAULT
475     mov dword ptr [rsp+48], 800
476     mov dword ptr [rsp+56], 600
477     call CreateWindowExA
478     mov g_hWnd, rax
479     
480     ; Show window
481     mov rcx, rax
482     mov edx, SW_SHOW
483     call ShowWindow
484     
485     mov rcx, g_hWnd
486     call UpdateWindow
487     
488     ; Message loop
489 MessageLoop:
490     lea rcx, msg
491     xor edx, edx
492     xor r8, r8
493     xor r9, r9
494     call GetMessageA
495     
496     test rax, rax
497     jz ExitMessageLoop
498     
499     lea rcx, msg
500     call TranslateMessage
501     
502     lea rcx, msg
503     call DispatchMessageA
504     
505     jmp MessageLoop
506     
507 ExitMessageLoop:
508     mov rax, msg.wParam
509     
510     add rsp, 64
511     pop rbp
512     ret
513 WinMain ENDP
